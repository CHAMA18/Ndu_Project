    82→DateTime? _parseFlexibleDate(String raw) {
    83→ final trimmed = raw.trim();
    84→ if (trimmed.isEmpty) return null;
    85→
    86→ final parsedIso = DateTime.tryParse(trimmed);
    87→ if (parsedIso != null) return parsedIso;
    88→
    89→ final mmdd = RegExp(r'^(\d{1,2})\/(\d{1,2})$').firstMatch(trimmed);
    90→ if (mmdd != null) {
    91→ final month = int.tryParse(mmdd.group(1) ?? '');
    92→ final day = int.tryParse(mmdd.group(2) ?? '');
    93→ if (month != null && day != null) {
    94→ return DateTime(DateTime.now().year, month, day);
    95→ }
    96→ }
    97→
    98→ final mmddyy =
    99→ RegExp(r'^(\d{1,2})\/(\d{1,2})\/(\d{2}|\d{4})$').firstMatch(trimmed);
   100→ if (mmddyy != null) {
   101→ final month = int.tryParse(mmddyy.group(1) ?? '');
   102→ final day = int.tryParse(mmddyy.group(2) ?? '');
   103→ final yearRaw = int.tryParse(mmddyy.group(3) ?? '');
   104→ if (month != null && day != null && yearRaw != null) {
   105→ final year = yearRaw < 100 ? 2000 + yearRaw : yearRaw;
   106→ return DateTime(year, month, day);
   107→ }
   108→ }
   109→
   110→ return null;
   111→}

   114→ if (value == null) return '';

   118→String _normalizedDateText(String raw, {bool fallbackToToday = false}) {
   119→ final parsed = _parseFlexibleDate(raw);
   120→ if (parsed != null) return _formatDate(parsed);
   121→ return fallbackToToday ? _formatDate(DateTime.now()) : '';

   136→ return showDatePicker(

   159→int? _durationDays(String start, String end) {
   160→ final s = _parseFlexibleDate(start);
   161→ final e = _parseFlexibleDate(end);
   162→ if (s == null || e == null) return null;
   163→ return e.difference(s).inDays.abs();

   389→List<ScheduleActivity> _syncQualityActivitiesToCalendar(
   390→ ProjectDataModel data,
   391→ QualityManagementData quality,
   392→) {
   393→ final updated = List<ScheduleActivity>.from(data.scheduleActivities);
   394→
   395→ void upsertActivity({
   396→ required String key,
   397→ required String title,
   398→ required String date,
   399→ required String assignee,
   400→ }) {
   401→ final normalizedDate = _normalizedDateText(date);
   402→ if (normalizedDate.isEmpty) return;
   403→ final entry = ScheduleActivity(
   404→ id: key,
   405→ title: title,
   406→ assignee: assignee,
   407→ startDate: normalizedDate,
   408→ dueDate: normalizedDate,
   409→ status: 'planned',
   410→ phase: 'execution',
   411→ workPackageType: 'execution',
   412→ discipline: 'Quality',
   413→ milestone: 'Quality',
   414→ durationDays: 1,
   415→ );
   416→ final index = updated.indexWhere(
   417→ (activity) => activity.id == key || activity.title == title,
   418→ );
   419→ if (index == -1) {
   420→ updated.add(entry);
   421→ } else {
   422→ updated[index] = entry;
   423→ }
   424→ }
   425→
   426→ for (final control in quality.workflowControls) {
   427→ upsertActivity(
   428→ key: 'quality-control-${control.id}',
   429→ title: 'Quality ${control.type.name.toUpperCase()} Review: ${control.name}',
   430→ date: control.frequency,
   431→ assignee: control.owner,
   432→ );
   433→ }
   434→ for (final audit in quality.auditPlan) {
   435→ upsertActivity(
   436→ key: 'quality-audit-${audit.id}',
   437→ title: 'Quality Audit: ${audit.title}',
   438→ date: audit.plannedDate,
   439→ assignee: audit.owner,
   440→ );
   441→ }
   442→ for (final task in [...quality.qaTaskLog, ...quality.qcTaskLog]) {
   443→ upsertActivity(
   444→ key: 'quality-task-${task.id}',
   445→ title: 'Quality Task: ${task.task}',
   446→ date: task.endDate.isNotEmpty ? task.endDate : task.startDate,
   447→ assignee: task.responsible,
   448→ );
   449→ }
   450→ for (final action in quality.correctiveActions) {
   451→ upsertActivity(
   452→ key: 'quality-corrective-${action.id}',
   453→ title: 'Corrective Action: ${action.title}',
   454→ date: action.dueDate,
   455→ assignee: action.owner,
   456→ );
   457→ }
   458→
   459→ return updated;
   460→}

  1205→ // ── Field history tracking for undo/redo ─────────────────────────────
  1206→ void _recordFieldHistory(String key, String value, {bool isAi = false}) {
  1207→ final history = _fieldHistories.putIfAbsent(key, () => []);
  1208→ final index = _fieldHistoryIndices.putIfAbsent(key, () => -1);
  1209→ if (index < history.length - 1) {
  1210→ history.removeRange(index + 1, history.length);
  1211→ }
  1212→ if (history.isEmpty || history.last != value) {
  1213→ history.add(value);
  1214→ _fieldHistoryIndices[key] = history.length - 1;
  1215→ }
  1216→ if (isAi) _fieldIsAiGenerated[key] = true;
  1217→ }
  1218→
  1219→ bool _canUndoField(String key) =>
  1220→ (_fieldHistoryIndices[key] ?? -1) > 0;
  1221→
  1222→ bool _canRedoField(String key) {
  1223→ final idx = _fieldHistoryIndices[key] ?? -1;
  1224→ final history = _fieldHistories[key] ?? [];
  1225→ return idx >= 0 && idx < history.length - 1;
  1226→ }
  1227→
  1228→ void _undoField(String key, TextEditingController controller) {
  1229→ if (!_canUndoField(key)) return;
  1230→ final idx = _fieldHistoryIndices[key]! - 1;
  1231→ _fieldHistoryIndices[key] = idx;
  1232→ controller.text = _fieldHistories[key]![idx];
  1233→ _onFieldChanged();
  1234→ }
  1235→
  1236→ void _redoField(String key, TextEditingController controller) {
  1237→ if (!_canRedoField(key)) return;
  1238→ final idx = _fieldHistoryIndices[key]! + 1;
  1239→ _fieldHistoryIndices[key] = idx;
  1240→ controller.text = _fieldHistories[key]![idx];
  1241→ _onFieldChanged();
  1242→ }

  1244→ // ── Per-field AI regeneration ────────────────────────────────────────
  1245→ Future<void> _regenerateField(String key, String label) async {
  1246→ setState(() => _fieldIsRegenerating[key] = true);
  1247→ try {
  1248→ final project = ProjectDataHelper.getData(context);
  1249→ final contextText =
  1250→ ProjectDataHelper.buildFepContext(project, sectionLabel: label);
  1251→ final ai = OpenAiServiceSecure();
  1252→ final result = await ai.generateCompletion(
  1253→ 'Based on this project context, regenerate the "$label" section.\n\n'
  1254→ 'Context:\n$contextText\n\n'
  1255→ 'Provide 2-3 sentences of specific, actionable recommendations. '
  1256→ 'Return ONLY the text content (no JSON, no markdown headers).',
  1257→ maxTokens: 300,
  1258→ temperature: 0.6,
  1259→ );
  1260→ final cleaned = result.trim();
  1261→ if (cleaned.isNotEmpty) {
  1262→ final controller = _controllerForKey(key);
  1263→ controller.text = cleaned;
  1264→ _recordFieldHistory(key, cleaned, isAi: true);
  1265→ _onFieldChanged();
  1266→ }
  1267→ } catch (e) {
  1268→ if (mounted) {
  1269→ ScaffoldMessenger.of(context).showSnackBar(
  1270→ SnackBar(content: Text('AI regeneration failed: $e')),
  1271→ );
  1272→ }
  1273→ }
  1274→ if (mounted) setState(() => _fieldIsRegenerating[key] = false);
  1275→ }
  1276→
  1277→ TextEditingController _controllerForKey(String key) {
  1278→ switch (key) {
  1279→ case 'quality_narrative':
  1280→ return _planController;
  1281→ case 'review_cadence':
  1282→ return _reviewCadenceController;
  1283→ case 'escalation_path':
  1284→ return _escalationPathController;
  1285→ case 'change_control':
