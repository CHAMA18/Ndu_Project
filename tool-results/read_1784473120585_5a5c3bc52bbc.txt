   193→  // ── Load saved state (visited tabs, stakeholder confirmation, plans, SDGs) ──
   194→  void _loadSavedState() {
   195→    final ssherData = ProjectDataHelper.getData(context).ssherData;
   196→    setState(() {
   197→      _stakeholderConfirmed = ssherData.stakeholderConfirmed;
   198→      _autoSyncToCostEstimate = ssherData.autoSyncToCostEstimate;
   199→      // Restore visited tabs from persistent storage
   200→      for (final cat in _SsherCategory.values) {
   201→        if (cat == _SsherCategory.cost) continue;
   202→        if (ssherData.categoryApplicability[cat.name] == false) {
   203→          _visitedTabs.add(cat);
   204→        }
   205→        if (ssherData.tabsVisited[cat.name] == true) {
   206→          _visitedTabs.add(cat);
   207→        }
   208→      }
   209→      // Restore per-category plans
   210→      for (final entry in ssherData.categoryPlans.entries) {
   211→        if (entry.value.trim().isNotEmpty) {
   212→          _categoryPlans[entry.key] = entry.value;
   213→          _categoryPlanLoaded[entry.key] = true;
   214→        }
   215→      }
   216→    });
   217→    // If auto-sync is enabled, mark all existing entries as already-synced
   218→    // so we don't retroactively push them on the first save.
   219→    if (_autoSyncToCostEstimate) {
   220→      for (final e in ssherData.entries) {
   221→        _autoSyncedEntryIds.add(e.id);
   222→      }
   223→    }
   224→  }

   226→  Future<void> _saveTabsVisited() async {
   227→    final visitedMap = <String, bool>{};
   228→    for (final cat in _SsherCategory.values) {
   229→      if (cat == _SsherCategory.cost) continue;
   230→      visitedMap[cat.name] = _visitedTabs.contains(cat);
   231→    }
   232→    await ProjectDataHelper.updateAndSave(
   233→      context: context,
   234→      checkpoint: 'ssher',
   235→      showSnackbar: false,
   236→      dataUpdater: (data) => data.copyWith(
   237→        ssherData: data.ssherData.copyWith(tabsVisited: visitedMap),
   238→      ),
   239→    );
   240→  }

   338→  /// Pull the Initiation phase (Front End Planning) security section data
   339→  /// into the Security tab so the user has continuity. Only runs once.
   340→  void _pullInitiationSecurityData() {
   341→    if (_initiationSecurityPulled) return;
   342→    final data = ProjectDataHelper.getData(context);
   343→    final fep = data.frontEndPlanning;
   344→    final securityText = fep.security.trim();
   345→
   346→    // Pull the security notes as a single entry on the Security tab if we don't already have one
   347→    final hasSecurityEntry = _securityEntries.any((e) =>
   348→        e.concern.toLowerCase().contains('initiation') ||
   349→        e.notes.toLowerCase().contains('initiation security') ||
   350→        e.concern == 'Initiation Phase Security Plan');
   351→
   352→    if (securityText.isNotEmpty && !hasSecurityEntry) {
   353→      // Summarize the initiation security text into a 200-char entry
   354→      final summary = securityText.length > 280
   355→          ? '${securityText.substring(0, 280)}...'
   356→          : securityText;
   357→      final entry = SsherEntry(
   358→        category: 'security',
   359→        department: 'IT Security',
   360→        teamMember: 'Security Analyst (carried from Initiation)',
   361→        concern: 'Initiation Phase Security Plan (carried forward)',
   362→        riskLevel: 'High',
   363→        mitigation: summary,
   364→        notes: 'Auto-imported from Front End Planning – Security section.',
   365→      );
   366→      setState(() {
   367→        _securityEntries.insert(0, entry);
   368→        _initiationSecurityPulled = true;
   369→      });
   370→      _saveEntries();
   371→    } else if (hasSecurityEntry) {
   372→      _initiationSecurityPulled = true;
   373→    }
   374→  }

   376→  Future<void> _populateSsherEntriesFromAi() async {
   377→    if (_entriesGenerated || _isGeneratingEntries) return;
   378→    if (_allEntries().isNotEmpty) {
   379→      _entriesGenerated = true;
   380→      return;
   381→    }
   382→
   383→    final projectData = ProjectDataHelper.getData(context);
   384→    final contextText =
   385→        ProjectDataHelper.buildFepContext(projectData, sectionLabel: 'SSHER');
   386→    if (contextText.trim().isEmpty) {
   387→      _entriesGenerated = true;
   388→      return;
   389→    }
   390→
   391→    setState(() => _isGeneratingEntries = true);
   392→
   393→    List<SsherEntry> generatedEntries = [];
   394→    try {
   395→      generatedEntries = await OpenAiServiceSecure()
   396→          .generateSsherEntries(context: contextText, itemsPerCategory: 2);
   397→    } catch (error) {
   398→      debugPrint('SSHER entries AI call failed: $error');
   399→    }
   400→
   401→    if (!mounted) return;
   402→
   403→    if (_allEntries().isNotEmpty) {
   404→      setState(() => _isGeneratingEntries = false);
   405→      _entriesGenerated = true;
   406→      return;
   407→    }
   408→
   409→    final safety = <SsherEntry>[];
   410→    final security = <SsherEntry>[];
   411→    final health = <SsherEntry>[];
   412→    final environment = <SsherEntry>[];
   413→    final regulatory = <SsherEntry>[];
   414→
   415→    for (final entry in generatedEntries) {
   416→      switch (entry.category) {
   417→        case 'safety':
   418→          safety.add(entry);
   419→          break;
   420→        case 'security':
   421→          security.add(entry);
   422→          break;
   423→        case 'health':
   424→          health.add(entry);
   425→          break;
   426→        case 'environment':
   427→          environment.add(entry);
   428→          break;
   429→        case 'regulatory':
   430→          regulatory.add(entry);
   431→          break;
   432→      }
   433→    }
   434→
   435→    setState(() {
   436→      _safetyEntries = safety;
   437→      _securityEntries = security;
   438→      _healthEntries = health;
   439→      _environmentEntries = environment;
   440→      _regulatoryEntries = regulatory;
   441→      _isGeneratingEntries = false;
   442→    });
   443→    _entriesGenerated = true;
   444→    await _saveEntries();
   445→    // After AI generation, also pull in Initiation security data
   446→    _pullInitiationSecurityData();
   447→  }

   449→  /// Ensures the per-category AI plan is loaded (lazy-loaded on first tab visit)
   450→  Future<void> _ensureCategoryPlanLoaded(_SsherCategory cat) async {
   451→    if (cat == _SsherCategory.cost) return;
   452→    final key = cat.name;
   453→    if (_categoryPlanLoaded[key] == true) return;
   454→    if (_categoryPlans[key]?.isNotEmpty == true) {
   455→      _categoryPlanLoaded[key] = true;
   456→      return;
   457→    }
   458→    if (_categoryPlanLoading[key] == true) return;
   459→
   460→    setState(() => _categoryPlanLoading[key] = true);
   461→
   462→    final projectData = ProjectDataHelper.getData(context);
   463→    final contextText =
   464→        ProjectDataHelper.buildFepContext(projectData, sectionLabel: 'SSHER');
   465→    if (contextText.trim().isEmpty) {
   466→      setState(() {
   467→        _categoryPlanLoaded[key] = true;
   468→        _categoryPlanLoading[key] = false;
   469→      });
   470→      return;
   471→    }
   472→
   473→    String plan = '';
   474→    try {
   475→      plan = await OpenAiServiceSecure().generateSsherCategoryPlan(
   476→        context: contextText,
   477→        category: key,
   478→      );
   479→    } catch (e) {
   480→      debugPrint('generateSsherCategoryPlan failed for $key: $e');
   481→    }
   482→
   483→    if (!mounted) return;
   484→
   485→    final trimmed = plan.trim();
   486→    setState(() {
   487→      if (trimmed.isNotEmpty) {
   488→        _categoryPlans[key] = trimmed;
   489→      }
   490→      _categoryPlanLoaded[key] = true;
   491→      _categoryPlanLoading[key] = false;
   492→    });
   493→
   494→    if (trimmed.isNotEmpty) {
   495→      await ProjectDataHelper.updateAndSave(
   496→        context: context,
   497→        checkpoint: 'ssher',
   498→        showSnackbar: false,
   499→        dataUpdater: (data) => data.copyWith(
   500→          ssherData: data.ssherData.copyWith(
   501→            categoryPlans: Map<String, String>.from(_categoryPlans),
   502→          ),
   503→        ),
   504→      );
   505→    }
   506→  }

   517→  /// Generates SDG recommendations (for the Environment tab)
   518→  Future<void> _loadSdgRecommendations() async {
   519→    if (_sdgsLoaded || _isGeneratingSdgs) return;
   520→    setState(() => _isGeneratingSdgs = true);
   521→
   522→    final projectData = ProjectDataHelper.getData(context);
   523→    final contextText =
   524→        ProjectDataHelper.buildFepContext(projectData, sectionLabel: 'SSHER');
   525→    if (contextText.trim().isEmpty) {
   526→      setState(() {
   527→        _isGeneratingSdgs = false;
   528→        _sdgsLoaded = true;
   529→      });
   530→      return;
   531→    }
   532→
   533→    List<Map<String, String>> recs = [];
   534→    try {
   535→      recs = await OpenAiServiceSecure()
   536→          .generateSdgRecommendations(context: contextText);
   537→    } catch (e) {
   538→      debugPrint('generateSdgRecommendations failed: $e');
   539→    }
   540→
   541→    if (!mounted) return;
   542→    setState(() {
   543→      _sdgRecommendations = recs;
   544→      _isGeneratingSdgs = false;
   545→      _sdgsLoaded = true;
   546→    });
   547→
   548→    if (recs.isNotEmpty) {
   549→      final sdgList = recs.map((r) => r['goal'] ?? '').where((g) => g.isNotEmpty).toList();
   550→      await ProjectDataHelper.updateAndSave(
   551→        context: context,
   552→        checkpoint: 'ssher',
   553→        showSnackbar: false,
   554→        dataUpdater: (data) => data.copyWith(
   555→          ssherData: data.ssherData.copyWith(applicableSdgs: sdgList),
   556→        ),
   557→      );
   558→    }
   559→  }
