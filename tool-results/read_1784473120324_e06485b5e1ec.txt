    23→  /// Capture the current project name so later modules (Cost Estimate,
    24→  /// Schedule) can read it without a [BuildContext].
    25→  static void captureProjectName(String? name) {
    26→    final trimmed = (name ?? '').trim();
    27→    if (trimmed.isEmpty) return;
    28→    _lastKnownProjectName = trimmed;
    29→  }

    31→  /// Read the project name from the [ProjectDataProvider] in [context], if
    32→  /// available, and update [lastKnownProjectName]. Returns the name (or null
    33→  /// if no provider is in scope or the name is empty).
    34→  static String? readProjectNameFromContext(BuildContext context) {
    35→    final provider = ProjectDataInherited.maybeOf(context);
    36→    if (provider == null) return _lastKnownProjectName;
    37→    final name = provider.projectData.projectName.trim();
    38→    if (name.isNotEmpty) {
    39→      _lastKnownProjectName = name;
    40→    }
    41→    return _lastKnownProjectName;
    42→  }

    44→  static ProjectMethodology? projectMethodologyFromOverallFramework(
    45→      String? framework) {
    46→    final normalized = (framework ?? '').trim().toLowerCase();
    47→    if (normalized == 'agile') return ProjectMethodology.agile;
    48→    if (normalized == 'hybrid') return ProjectMethodology.hybrid;
    49→    if (normalized == 'waterfall') return ProjectMethodology.waterfall;
    50→    return null;
    51→  }

    73→  /// Check if a destination checkpoint is locked/not accessible
    74→  /// Returns true if the destination is locked, false if accessible
    75→  static bool isDestinationLocked(
    76→      BuildContext context, String destinationCheckpoint) {
    77→    final provider = Provider.of<ProjectDataProvider>(context, listen: false);
    78→
    79→    final projectData = provider.projectData;
    80→    final currentCheckpoint = projectData.currentCheckpoint;
    81→
    82→    // Check if it's a Basic Plan locked item
    83→    const basicPlanLockedCheckpoints = {
    84→      'fep_contract_vendor_quotes',
    85→      'fep_security',
    86→      'fep_allowance',
    87→      'work_breakdown_structure',
    88→      'interface_management',
    89→      'project_baseline',
    90→      'project_plan_level1_schedule',
    91→      'project_plan_detailed_schedule',
    92→      'project_plan_condensed_summary',
    93→      'team_management',
    94→      'staff_team',
    95→      'update_ops_maintenance_plans',
    96→      'gap_analysis_scope_reconcillation',
    97→      'punchlist_actions',
    98→      'salvage_disposal_team',
    99→      'engineering_design',
   100→      'specialized_design',
   101→      'technical_development',
   102→      'project_summary',
   103→      'warranties_operations_support',
   104→      'project_financial_review',
   105→    };
   106→
   107→    if (projectData.isBasicPlanProject &&
   108→        basicPlanLockedCheckpoints.contains(destinationCheckpoint)) {
   109→      return true;
   110→    }
   111→
   112→    // Check if checkpoint has been reached
   113→    if (currentCheckpoint.isEmpty) {
   114→      // Only allow first checkpoint if no progress
   115→      return destinationCheckpoint !=
   116→          SidebarNavigationService.instance.getNextItem(null)?.checkpoint;
   117→    }
   118→
   119→    return !SidebarNavigationService.instance
   120→        .isCheckpointReached(destinationCheckpoint, currentCheckpoint);
   121→  }

   154→  /// Save current screen data and navigate to next screen with automatic Firebase sync
   155→  /// Includes security check to prevent navigation to locked destinations
   156→  static Future<void> saveAndNavigate({
   157→    required BuildContext context,
   158→    required String checkpoint,
   159→    required Widget Function() nextScreenBuilder,
   160→    ProjectDataModel Function(ProjectDataModel)? dataUpdater,
   161→    String?
   162→        destinationCheckpoint, // Optional: checkpoint of destination screen for lock checking
   163→    String? destinationName, // Optional: human-readable name for error messages
   164→    bool saveInBackground = false, // Opt-in: navigate immediately, save async
   165→  }) async {
   166→    final provider = Provider.of<ProjectDataProvider>(context, listen: false);
   167→
   168→    // Security check: Verify destination is not locked
   169→    if (destinationCheckpoint != null &&
   170→        isDestinationLocked(context, destinationCheckpoint)) {
   171→      showLockedDestinationMessage(context, destinationName ?? 'the next page');
   172→      return; // Block navigation
   173→    }
   174→
   175→    // Update data if updater is provided
   176→    if (dataUpdater != null) {
   177→      provider.updateField(dataUpdater);
   178→    }
   179→
   180→    if (saveInBackground) {
   181→      // Update checkpoint in-memory so downstream widgets can reflect progress
   182→      // immediately without waiting on network IO.
   183→      provider
   184→          .updateField((data) => data.copyWith(currentCheckpoint: checkpoint));
   185→
   186→      // Navigate immediately to reduce perceived latency.
   187→      if (context.mounted) {
   188→        PhaseTransitionHelper.pushPhaseAware(
   189→          context: context,
   190→          builder: (_) => nextScreenBuilder(),
   191→          destinationCheckpoint: destinationCheckpoint,
   192→          sourceCheckpoint: checkpoint,
   193→        );
   194→      }
   195→
   196→      // Save in the background (no UI blocking). We intentionally avoid showing
   197→      // snackbars here because the source context may be disposed after nav.
   198→      Future<void>(() async {
   199→        try {
   200→          final success = await provider.saveToFirebase(checkpoint: checkpoint);
   201→          if (!success) {
   202→            debugPrint(
   203→              'Warning: ${provider.lastError ?? "Could not save data"} (background save)',
   204→            );
   205→          }
   206→        } catch (e) {
   207→          debugPrint('Background save error: $e');
   208→        }
   209→      });
   210→
   211→      return;
   212→    }
   213→
   214→    // Save to Firebase (blocking)
   215→    final success = await provider.saveToFirebase(checkpoint: checkpoint);
   216→
   217→    if (!success && context.mounted) {
   218→      ScaffoldMessenger.of(context).showSnackBar(
   219→        SnackBar(
   220→          content:
   221→              Text('Warning: ${provider.lastError ?? "Could not save data"}'),
   222→          backgroundColor: Colors.orange,
   223→        ),
   224→      );
   225→    }
   226→
   227→    // Navigate to next screen
   228→    if (context.mounted) {
   229→      PhaseTransitionHelper.pushPhaseAware(
   230→        context: context,
   231→        builder: (_) => nextScreenBuilder(),
   232→        destinationCheckpoint: destinationCheckpoint,
   233→        sourceCheckpoint: checkpoint,
   234→      );
   235→    }
   236→  }

   238→  /// Build a compact, structured context string for Front End Planning prompts.
   239→  /// This aggregates prior inputs across the project to enable high‑quality AI suggestions.
   240→  ///
   241→  /// When [wbs] and/or [costEstimate] are provided, the scan is enriched with
   242→  /// a WBS structure section (framework + Level 1/2 node names) and a Cost
   243→  /// Estimate summary (total + top cost lines), so AI prompts on later pages
   244→  /// (Cost Estimate, Schedule, etc.) can reference the WBS and cost data.
   245→  static String buildProjectContextScan(
   246→    ProjectDataModel data, {
   247→    String? sectionLabel,
   248→    WBS? wbs,
   249→    CostEstimate? costEstimate,
   250→  }) {
   251→    // Capture the latest project name so modules without a BuildContext
   252→    // (e.g. CostEstimateProvider.setup) can read it later.
   253→    captureProjectName(data.projectName);
   254→
   255→    final enriched = ProjectIntelligenceService.rebuildActivityLog(data);
   256→    final base = ProjectIntelligenceService.buildContextScan(
   257→      enriched,
   258→      sectionLabel: sectionLabel,
   259→    );
   260→
   261→    final wbsSection = wbs != null ? _wbsContextSection(wbs) : '';
   262→    final costSection =
   263→        costEstimate != null ? _costEstimateContextSection(costEstimate) : '';
   264→
   265→    final parts = <String>[
   266→      base,
   267→      if (wbsSection.isNotEmpty) wbsSection,
   268→      if (costSection.isNotEmpty) costSection,
   269→    ];
   270→    return parts.where((s) => s.trim().isNotEmpty).join('\n\n');
   271→  }

   273→  /// Build a compact WBS summary section for inclusion in AI context scans.
   274→  /// Shows the framework, project name, and Level 1 / Level 2 node names so
   275→  /// downstream AI prompts can reference the WBS structure.
   276→  static String _wbsContextSection(WBS wbs) {
   277→    final buf = StringBuffer();
   278→    final counts = countNodes(wbs);
   279→    buf.writeln('Work Breakdown Structure');
   280→    buf.writeln('------------------------');
   281→    buf.writeln('Project: ${wbs.projectName}');
   282→    buf.writeln('Framework: ${wbs.framework.label}');
   283→    buf.writeln(
   284→        'Levels: ${counts.level1} ${wbs.framework.level1Label} · ${counts.level2} ${wbs.framework.level2Label}');
   285→    if (wbs.level0.children.isNotEmpty) {
   286→      buf.writeln('${wbs.framework.level1Label}:');
   287→      for (final l1 in wbs.level0.children) {
   288→        buf.writeln('- ${l1.code} ${l1.name}');
   289→        if (l1.children.isNotEmpty) {
   290→          for (final l2 in l1.children) {
   291→            buf.writeln('    · ${l2.code} ${l2.name}');
   292→          }
   293→        }
   294→      }
   295→    }
   296→    return buf.toString().trim();
   297→  }

   299→  /// Build a compact Cost Estimate summary section for inclusion in AI
   300→  /// context scans. Shows the project name, estimate class, total, and the
   301→  /// top cost lines by amount so later pages can reference cost data.
   302→  static String _costEstimateContextSection(CostEstimate estimate) {
   303→    final buf = StringBuffer();
   304→    buf.writeln('Cost Estimate');
   305→    buf.writeln('-------------');
   306→    buf.writeln('Project: ${estimate.projectName}');
   307→    buf.writeln('Class: ${estimate.className.label} (${estimate.className.name})');
   308→    buf.writeln('Delivery model: ${estimate.deliveryModel.label}');
   309→    buf.writeln('Status: ${estimate.status.label}');
   310→    buf.writeln('Currency: ${estimate.currency}');
   311→    final total = estimate.lines.fold<double>(
   312→        0, (s, l) => s + _effectiveLineTotalForContext(l));
   313→    buf.writeln('Total: ${total.toStringAsFixed(2)} ${estimate.currency}');
   314→    buf.writeln('Lines: ${estimate.lines.length}');
   315→    if (estimate.lines.isNotEmpty) {
   316→      final sorted = [...estimate.lines]
   317→        ..sort((a, b) => _effectiveLineTotalForContext(b)
   318→            .compareTo(_effectiveLineTotalForContext(a)));
   319→      final top = sorted.take(8).toList();
   320→      buf.writeln('Top cost lines:');
   321→      for (final l in top) {
   322→        final wbsRef = (l.wbsRef ?? '').trim();
   323→        final refSuffix = wbsRef.isEmpty ? '' : ' [WBS: $wbsRef]';
   324→        buf.writeln(
   325→            '- ${l.category.label} · ${l.description}$refSuffix · ${_effectiveLineTotalForContext(l).toStringAsFixed(2)} ${estimate.currency}');
   326→      }
   327→    }
   328→    return buf.toString().trim();
   329→  }

   331→  /// Mirror of [ComputeUtils] effective line total — kept private to avoid
   332→  /// importing the cost estimate compute utils from this helper (which would
   333→  /// create a circular dependency in some downstream import graphs).
   334→  static double _effectiveLineTotalForContext(CostLine l) {
   335→    if (l.varianceType == VarianceType.remove) {
   336→      return -(l.varianceBaselineTotal ?? 0);
   337→    }
   338→    if (l.varianceType == VarianceType.change) {
   339→      return l.varianceDelta ?? 0;
   340→    }
   341→    return l.total;
   342→  }

   344→  static List<String> _formatInterfaceEntriesForContext(
   345→      List<InterfaceEntry> entries) {
   346→    final formatted = <String>[];
   347→    for (final entry in entries) {
   348→      final boundary = entry.boundary.trim().isNotEmpty
   349→          ? entry.boundary.trim()
   350→          : 'Unnamed interface';
   351→      final details = [
   352→        if (entry.owner.trim().isNotEmpty) 'Owner: ${entry.owner.trim()}',
   353→        if (entry.cadence.trim().isNotEmpty) 'Cadence: ${entry.cadence.trim()}',
   354→        if (entry.risk.trim().isNotEmpty) 'Risk: ${entry.risk.trim()}',
   355→        if (entry.status.trim().isNotEmpty) 'Status: ${entry.status.trim()}',
   356→        if (entry.lastSync.trim().isNotEmpty)
   357→          'Last sync: ${entry.lastSync.trim()}',
   358→        if (entry.notes.trim().isNotEmpty) 'Notes: ${entry.notes.trim()}',
   359→      ].join(' | ');
   360→      final entryText = details.isNotEmpty ? '$boundary | $details' : boundary;
   361→      formatted.add(entryText);
   362→    }
   363→    return formatted;
   364→  }

   366→  static String buildFepContext(ProjectDataModel data, {String? sectionLabel}) {
   367→    final buf = StringBuffer();
   368→    void w(String label, String? value) {
   369→      final v = (value ?? '').trim();
   370→      if (v.isEmpty) return;
   371→      buf.writeln('$label:');
   372→      buf.writeln(v);
   373→      buf.writeln();
   374→    }
   375→
   376→    buf.writeln('Project Context');
   377→    buf.writeln('================');
   378→    w('Project Name', data.projectName);
   379→    w('Solution Title', data.solutionTitle);
   380→    w('Solution Description', data.solutionDescription);
   381→    w('Business Case', data.businessCase);
   382→    w('Initiation Notes', data.notes);
   383→    w('Potential Solution', data.potentialSolution);
   384→    w('Project Objective', data.projectObjective);
   385→    w('Overall Framework', data.overallFramework);
   386→
   387→    if (data.projectGoals.isNotEmpty) {
   388→      buf.writeln('Project Goals:');
   389→      for (final g in data.projectGoals) {
   390→        final name = (g.name).trim();
   391→        final desc = (g.description).trim();
   392→        if (name.isEmpty && desc.isEmpty) continue;
   393→        buf.writeln(
   394→            '- ${name.isEmpty ? 'Goal' : name}: ${desc.isEmpty ? '' : desc}');
   395→      }
   396→      buf.writeln();
   397→    }
   398→
   399→    if (data.planningGoals.isNotEmpty) {
   400→      buf.writeln('Planning Goals:');
   401→      for (final g in data.planningGoals) {
   402→        final title = (g.title).trim();
   403→        final desc = (g.description).trim();
   404→        final year = (g.targetYear).trim();
   405→        if (title.isEmpty && desc.isEmpty && year.isEmpty) continue;
   406→        buf.writeln(
   407→            '- ${title.isEmpty ? 'Goal ${g.goalNumber}' : title} (${year.isEmpty ? 'n/a' : year}): $desc');
   408→      }
   409→      buf.writeln();
   410→    }
   411→
   412→    if (data.keyMilestones.isNotEmpty) {
   413→      buf.writeln('Key Milestones:');
   414→      for (final m in data.keyMilestones) {
   415→        final name = (m.name).trim();
   416→        final due = (m.dueDate).trim();
   417→        final discipline = (m.discipline).trim();
   418→        if (name.isEmpty && due.isEmpty && discipline.isEmpty) continue;
   419→        buf.writeln(
   420→            '- ${name.isEmpty ? 'Milestone' : name} | Due: ${due.isEmpty ? 'TBD' : due} | ${discipline.isEmpty ? '' : 'Discipline: $discipline'}');
   421→      }
   422→      buf.writeln();
   423→    }
   424→
   425→    if (data.planningNotes.isNotEmpty) {
   426→      buf.writeln('Planning Phase Notes:');
   427→      data.planningNotes.forEach((key, value) {
   428→        final v = value.trim();
   429→        if (v.isEmpty) return;
   430→        buf.writeln('- ${key.trim()}: $v');
   431→      });
   432→      buf.writeln();
   433→    }
   434→
   435→    // Include any prior Front End Planning fields already provided
   436→    final fep = data.frontEndPlanning;
   437→    w('Front End Planning – Requirements Notes', fep.requirementsNotes);
   438→    w('Front End Planning – Requirements', fep.requirements);
   439→    w('Front End Planning – Risks', fep.risks);
   440→    w('Front End Planning – Opportunities', fep.opportunities);
   441→    w('Front End Planning – Contracting', fep.contractVendorQuotes);
   442→    w('Front End Planning – Procurement', fep.procurement);
   443→    w('Front End Planning – Security', fep.security);
   444→    w('Front End Planning – Allowance', fep.allowance);
   445→    w('Front End Planning – Summary', fep.summary);
   446→    w('Front End Planning – Technology', fep.technology);
   447→    w('Front End Planning – Personnel', fep.personnel);
   448→    w('Front End Planning – Infrastructure', fep.infrastructure);
   449→
   450→    final interfaceEntriesContext =
   451→        _formatInterfaceEntriesForContext(data.interfaceEntries);
   452→    if (interfaceEntriesContext.isNotEmpty) {
   453→      buf.writeln('Interface Register:');
   454→      for (final entry in interfaceEntriesContext) {
   455→        buf.writeln('- $entry');
   456→      }
   457→      buf.writeln();
   458→    }
   459→
   460→    if ((sectionLabel ?? '').isNotEmpty) {
   461→      buf.writeln('Target Section: ${sectionLabel!.trim()}');
   462→    }
   463→
   464→    return buf.toString().trim();
   465→  }

   467→  /// Build context for AI-generated Project Objective summaries.
   468→  /// Includes charter inputs and front-end planning details without reusing
   469→  /// an existing objective to avoid circular summaries.
   470→  static String buildProjectObjectiveContext(ProjectDataModel data) {
   471→    final buf = StringBuffer();
   472→
   473→    void w(String label, String? value) {
   474→      final v = (value ?? '').trim();
   475→      if (v.isEmpty) return;
   476→      buf.writeln('$label:');
   477→      buf.writeln(v);
   478→      buf.writeln();
   479→    }
   480→
   481→    void wList(String label, Iterable<String> items) {
   482→      final list =
   483→          items.map((e) => e.trim()).where((e) => e.isNotEmpty).toList();
   484→      if (list.isEmpty) return;
   485→      buf.writeln('$label:');
   486→      for (final item in list) {
   487→        buf.writeln('- $item');
   488→      }
   489→      buf.writeln();
   490→    }
   491→
   492→    buf.writeln('Project Context');
   493→    buf.writeln('================');
   494→    w('Project Name', data.projectName);
   495→    w('Solution Title', data.solutionTitle);
   496→    w('Solution Description', data.solutionDescription);
   497→    w('Business Case', data.businessCase);
   498→    w('Initiation Notes', data.notes);
   499→    w('Potential Solution', data.potentialSolution);
   500→    w('Overall Framework', data.overallFramework);
   501→
   502→    w('Charter Assumptions', data.charterAssumptions);
   503→    w('Charter Constraints', data.charterConstraints);
   504→
   505→    if (data.projectGoals.isNotEmpty) {
   506→      final items = data.projectGoals.map((g) {
   507→        final name = g.name.trim().isEmpty ? 'Goal' : g.name.trim();
   508→        final desc = g.description.trim();
   509→        return desc.isEmpty ? name : '$name: $desc';
   510→      });
   511→      wList('Project Goals', items);
   512→    }
   513→
   514→    if (data.planningGoals.isNotEmpty) {
   515→      final items = data.planningGoals.map((g) {
   516→        final title =
   517→            g.title.trim().isEmpty ? 'Goal ${g.goalNumber}' : g.title.trim();
   518→        final year = g.targetYear.trim();
   519→        final desc = g.description.trim();
   520→        final suffix = [
   521→          if (year.isNotEmpty) 'Target: $year',
   522→          if (desc.isNotEmpty) desc,
   523→        ].join(' | ');
   524→        return suffix.isEmpty ? title : '$title ($suffix)';
   525→      });
   526→      wList('Planning Goals', items);
   527→    }
   528→
   529→    if (data.keyMilestones.isNotEmpty) {
   530→      final items = data.keyMilestones.map((m) {
   531→        final name = m.name.trim().isNotEmpty ? m.name.trim() : 'Milestone';
   532→        final due = m.dueDate.trim();
   533→        final discipline = m.discipline.trim();
   534→        final details = [
   535→          if (due.isNotEmpty) 'Due: $due',
   536→          if (discipline.isNotEmpty) 'Discipline: $discipline',
   537→        ].join(' | ');
   538→        return details.isEmpty ? name : '$name ($details)';
   539→      });
   540→      wList('Key Milestones', items);
   541→    }
   542→
   543→    wList('Within Scope', data.withinScopeItems.map((e) => e.description));
   544→    wList('Out of Scope', data.outOfScopeItems.map((e) => e.description));
   545→    wList('Assumptions', data.assumptionItems.map((e) => e.description));
   546→    wList('Constraints', data.constraintItems.map((e) => e.description));
   547→
   548→    if (data.planningNotes.isNotEmpty) {
   549→      final items = data.planningNotes.entries

   583→  /// Build a richer, cross-application context string for executive plan diagrams.
   584→  /// Includes only populated fields to avoid noise and random output.
   585→  static String buildExecutivePlanContext(ProjectDataModel data,
   586→      {String? sectionLabel}) {
   587→    final buf = StringBuffer();
   588→    var hasContent = false;
   589→
   590→    String clamp(String value, {int max = 420}) {
   591→      final trimmed = value.trim();
   592→      if (trimmed.length <= max) return trimmed;
   593→      return '${trimmed.substring(0, max - 3)}...';
   594→    }
   595→
   596→    void w(String label, String? value) {
   597→      final v = clamp(value ?? '');
   598→      if (v.isEmpty) return;
   599→      hasContent = true;
   600→      buf.writeln('$label:');
   601→      buf.writeln(v);
   602→      buf.writeln();
   603→    }
   604→
   605→    void wList(String label, Iterable<String> items) {
   606→      final list = items.map(clamp).where((e) => e.isNotEmpty).toList();
   607→      if (list.isEmpty) return;
   608→      hasContent = true;
   609→      buf.writeln('$label:');
   610→      for (final item in list) {
   611→        buf.writeln('- $item');
   612→      }
   613→      buf.writeln();
   614→    }
   615→
   616→    buf.writeln('Project Context');
   617→    buf.writeln('================');
   618→    w('Project Name', data.projectName);
   619→    w('Solution Title', data.solutionTitle);
   620→    w('Solution Description', data.solutionDescription);
   621→    w('Business Case', data.businessCase);
   622→    w('Project Objective', data.projectObjective);
   623→    w('Overall Framework', data.overallFramework);
   624→    w('Notes', data.notes);
   625→    wList('Tags', data.tags);
   626→

   628→      final items = data.projectGoals.map((g) {
   629→        final name = g.name.trim().isEmpty ? 'Goal' : g.name.trim();

   639→            g.title.trim().isEmpty ? 'Goal ${g.goalNumber}' : g.title.trim();

  1009→  // ── Bidirectional Risk Register Sync ──
  1010→  //
  1011→  // Section-specific risks (e.g. Interface Management, SSHER, Design) are
  1012→  // mirrored into the central `frontEndPlanning.riskRegisterItems` list with
  1013→  // a `sourceSection` tag. Updates to either side propagate to the other.
  1014→
  1015→  /// Upserts a risk into the central Risk Register, tagged with its source
  1016→  /// section via `requirementType`. The `category` field is preserved for
  1017→  /// user/discipline classification (e.g. 'Safety', 'Contractual').
  1018→  /// De-duplication matches on `riskName + requirementType` (the sourceSection).
  1019→  static Future<void> upsertRiskToRegister({
  1020→    required BuildContext context,
  1021→    required String sourceSection,
  1022→    required String riskName,
  1023→    String description = '',
  1024→    String category = '',
  1025→    String impactLevel = 'Medium',
  1026→    String likelihood = 'Medium',
  1027→    String mitigationStrategy = '',
  1028→    String discipline = '',
  1029→    String owner = '',
  1030→    String status = 'Open',
  1031→    String checkpoint = 'risk_register_sync',
  1032→  }) async {
  1033→    await updateAndSave(
  1034→      context: context,
  1035→      checkpoint: checkpoint,
  1036→      showSnackbar: false,
  1037→      dataUpdater: (d) {
  1038→        final items = List<RiskRegisterItem>.from(
  1039→            d.frontEndPlanning.riskRegisterItems);
  1040→        final idx = items.indexWhere((r) =>
  1041→            r.riskName.trim().toLowerCase() == riskName.trim().toLowerCase() &&
  1042→            r.requirementType.trim().toLowerCase() ==
  1043→                sourceSection.trim().toLowerCase());
  1044→        if (idx >= 0) {
  1045→          // Update existing — preserve category if caller didn't provide one
  1046→          items[idx] = RiskRegisterItem(
  1047→            riskName: riskName,
  1048→            description: description.isNotEmpty ? description : items[idx].description,
  1049→            category: category.isNotEmpty ? category : items[idx].category,
  1050→            requirement: items[idx].requirement,
  1051→            requirementType: sourceSection,
  1052→            impactLevel: impactLevel,
  1053→            likelihood: likelihood,
  1054→            mitigationStrategy: mitigationStrategy.isNotEmpty
  1055→                ? mitigationStrategy
  1056→                : items[idx].mitigationStrategy,
  1057→            discipline: discipline.isNotEmpty ? discipline : items[idx].discipline,
  1058→            projectRole: items[idx].projectRole,
  1059→            owner: owner.isNotEmpty ? owner : items[idx].owner,
  1060→            status: status,
  1061→            probabilityNumeric: items[idx].probabilityNumeric,
  1062→            costImpactMin: items[idx].costImpactMin,
  1063→            costImpactMostLikely: items[idx].costImpactMostLikely,
  1064→            costImpactMax: items[idx].costImpactMax,
  1065→            scheduleImpactMin: items[idx].scheduleImpactMin,
  1066→            scheduleImpactMostLikely: items[idx].scheduleImpactMostLikely,
  1067→            scheduleImpactMax: items[idx].scheduleImpactMax,
  1068→            controlAccountId: items[idx].controlAccountId,
  1069→            cbsId: items[idx].cbsId,
  1070→            riskType: items[idx].riskType,
  1071→            responseStrategy: items[idx].responseStrategy,
  1072→            residualProbability: items[idx].residualProbability,
  1073→            residualCostImpact: items[idx].residualCostImpact,
  1074→          );
  1075→        } else {
  1076→          items.add(RiskRegisterItem(
  1077→            riskName: riskName,
  1078→            description: description,
  1079→            category: category.isNotEmpty ? category : sourceSection,
  1080→            requirementType: sourceSection,
  1081→            impactLevel: impactLevel,
  1082→            likelihood: likelihood,
  1083→            mitigationStrategy: mitigationStrategy,
  1084→            discipline: discipline,
  1085→            owner: owner,
  1086→            status: status,
  1087→          ));
  1088→        }
  1089→        return d.copyWith(
  1090→          frontEndPlanning:
  1091→              d.frontEndPlanning.copyWith(riskRegisterItems: items),
  1092→        );
  1093→      },
  1094→    );

  1211→    required BuildContext context,
  1212→    required String sourceSection,
  1213→    required String title,
  1214→    String description = '',
  1215→    String phase = 'Planning',
  1216→    String discipline = '',

  1517→  /// Unified total estimated cost used by Project Charter.
  1518→  static double getTotalEstimatedCostValue(ProjectDataModel data) {
  1519→    final forecastTotal =
  1520→        getCostEstimateTotalByState(data, costState: 'forecast');
  1521→    if (forecastTotal > 0) return forecastTotal;
  1522→
  1523→    final operationalTotal = data.contractors
  1524→            .fold<double>(0.0, (sum, item) => sum + item.estimatedCost) +
  1525→        data.vendors

  1527→        data.frontEndPlanning.allowanceItems
  1528→            .fold<double>(0.0, (sum, item) => sum + item.amount);
  1529→    if (operationalTotal > 0) return operationalTotal;
  1530→
  1531→    final costAnalysisTotal = data.costAnalysisData?.solutionCosts.fold<double>(
  1532→            0.0,
  1533→            (sum, solution) =>
  1534→                sum +
  1535→                solution.costRows.fold<double>(0.0,
  1536→                    (rowSum, row) => rowSum + _parseNumericValue(row.cost))) ??
  1537→        0.0;
  1538→
  1539→    return costAnalysisTotal;
  1540→  }

  1571→  static List<Milestone> _mergeAutoScheduleMilestones(
  1572→    List<Milestone> current,
  1573→    FrontEndPlanningData fep,
  1574→  ) {
  1575→    final manual = current
  1576→        .where((m) => !m.comments.contains(_autoScheduleMarker))
  1577→        .toList();
  1578→
  1579→    final generated = <Milestone>[];
  1580→
  1581→    for (final opp in fep.opportunityItems) {
  1582→      if (!_hasTag(opp.appliesTo, 'Schedule')) continue;
  1583→      final title =
  1584→          _withFallback(opp.opportunity, 'Opportunity ${generated.length + 1}');
  1585→      final scheduleSavings = opp.potentialScheduleSavings.trim();
  1586→      final marker = '$_autoScheduleMarker | opp:${opp.id}';
  1587→      generated.add(
  1588→        Milestone(
  1589→          name: 'Opportunity: $title',
  1590→          discipline: _withFallback(opp.discipline, 'Planning'),
  1591→          dueDate: '',
  1592→          comments: _buildAutoNote(
  1593→            source: marker,
  1594→            owner: opp.assignedTo,
  1595→            extra: scheduleSavings.isNotEmpty
  1596→                ? 'Potential schedule savings: $scheduleSavings'
  1597→                : null,
  1598→          ),
  1599→        ),
  1600→      );
  1601→    }
  1602→
  1603→    for (final allowance in fep.allowanceItems) {
  1604→      if (!_hasTag(allowance.appliesTo, 'Schedule')) continue;
  1605→      if (allowance.name.trim().isEmpty && allowance.notes.trim().isEmpty) {
  1606→        continue;
  1607→      }
  1608→      final title = _withFallback(
  1609→          allowance.name, 'Allowance ${allowance.number.toString()}');
  1610→      final marker = '$_autoScheduleMarker | allow:${allowance.id}';
  1611→      generated.add(
  1612→        Milestone(
  1613→          name: 'Allowance: $title',
  1614→          discipline: _withFallback(allowance.type, 'Planning'),
  1615→          dueDate: '',
  1616→          comments: _buildAutoNote(
  1617→            source: marker,
  1618→            owner: allowance.assignedTo,
  1619→            extra: allowance.notes.trim().isNotEmpty ? allowance.notes : null,
  1620→          ),
  1621→        ),
  1622→      );
  1623→    }
  1624→
  1625→    return [...manual, ...generated];
  1626→  }

  1628→  static List<TrainingActivity> _mergeAutoTrainingActivities(
  1629→    List<TrainingActivity> current,
  1630→    FrontEndPlanningData fep,
  1631→  ) {
  1632→    final manual = current
  1633→        .where((activity) =>

  1783→  static List<CostEstimateItem> _reconcileCostEstimateItems(
  1784→      List<CostEstimateItem> items) {
  1785→    final grouped = <String, List<CostEstimateItem>>{};
  1786→    final passthrough = <CostEstimateItem>[];
  1787→
  1788→    for (final item in items) {
  1789→      if (item.isBaseline || !_isAutoImportedCostSource(item.source)) {
  1790→        passthrough.add(item);
  1791→        continue;
  1792→      }
  1793→      final key = _reconciliationGroupKey(item);
  1794→      if (key == null) {
  1795→        passthrough.add(item);
  1796→        continue;
  1797→      }
  1798→      grouped.putIfAbsent(key, () => <CostEstimateItem>[]).add(item);
  1799→    }
  1800→
  1801→    final reconciled = <CostEstimateItem>[...passthrough];
  1802→    for (final groupItems in grouped.values) {
  1803→      final highestPriority = groupItems
  1804→          .map(_costStatePriority)
  1805→          .fold<int>(0, (highest, value) => value > highest ? value : highest);
  1806→      reconciled.addAll(groupItems.where(
  1807→        (item) => _costStatePriority(item) == highestPriority,
  1808→      ));
  1809→    }
  1810→    return reconciled;
  1811→  }

  1856→        return item.id.substring(prefix.length);
