    84‚ÜíDateTime? _parseFlexibleDate(String raw) {
    85‚Üí final trimmed = raw.trim();
    86‚Üí if (trimmed.isEmpty) return null;
    87‚Üí
    88‚Üí final parsedIso = DateTime.tryParse(trimmed);
    89‚Üí if (parsedIso != null) return parsedIso;
    90‚Üí
    91‚Üí final mmdd = RegExp(r'^(\d{1,2})\/(\d{1,2})$').firstMatch(trimmed);
    92‚Üí if (mmdd != null) {
    93‚Üí final month = int.tryParse(mmdd.group(1) ?? '');
    94‚Üí final day = int.tryParse(mmdd.group(2) ?? '');
    95‚Üí if (month != null && day != null) {
    96‚Üí return DateTime(DateTime.now().year, month, day);
    97‚Üí }
    98‚Üí }
    99‚Üí
   100‚Üí final mmddyy =
   101‚Üí RegExp(r'^(\d{1,2})\/(\d{1,2})\/(\d{2}|\d{4})$').firstMatch(trimmed);
   102‚Üí if (mmddyy != null) {
   103‚Üí final month = int.tryParse(mmddyy.group(1) ?? '');
   104‚Üí final day = int.tryParse(mmddyy.group(2) ?? '');
   105‚Üí final yearRaw = int.tryParse(mmddyy.group(3) ?? '');
   106‚Üí if (month != null && day != null && yearRaw != null) {
   107‚Üí final year = yearRaw < 100 ? 2000 + yearRaw : yearRaw;
   108‚Üí return DateTime(year, month, day);
   109‚Üí }
   110‚Üí }
   111‚Üí
   112‚Üí return null;
   113‚Üí}
