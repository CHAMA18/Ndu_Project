   120→ Future<void> _regenerateCurrentTab() async {
   121→ if (_regenerating) return;
   122→ final provider = ProjectDataInherited.maybeOf(context);
   123→ if (provider == null) return;
   124→
   125→ setState(() => _regenerating = true);
   126→ final ai = OpenAiServiceSecure();
   127→ final ctx = _buildAiContextForCurrentTab(provider.projectData);
   128→
   129→ try {
   130→ switch (_selectedTab) {
   131→ case _TechnologyTab.inventory:
   132→ final text = await ai.generateFepSectionText(
   133→ section: 'Technology Inventory',
   134→ context:
   135→ '$ctx\nOutput format (one per line): Name | Category | Cost | Status | Vendor | Added Date (YYYY-MM-DD) | Notes',
   136→ maxTokens: 700,
   137→ );
   138→ final lines = TextSanitizer.sanitizeAiText(text)
   139→ .split('\n')
   140→ .where((l) => l.trim().isNotEmpty);
   141→ _inventory
   142→ ..clear()
   143→ ..addAll(lines.map((line) {
   144→ final parts = line.split('|').map((p) => p.trim()).toList();
   145→ return {
   146→ 'name': parts.isNotEmpty ? parts[0] : line.trim(),
   147→ 'category': parts.length > 1 ? parts[1] : 'Software',
   148→ 'cost': parts.length > 2 ? parts[2] : '',
   149→ 'status': _normalizeStatus(
   150→ parts.length > 3 ? parts[3] : 'Proposed/Pending'),
   151→ 'vendor': parts.length > 4 ? parts[4] : '',
   152→ 'added': parts.length > 5 ? parts[5] : '',
   153→ 'notes': parts.length > 6 ? parts.sublist(6).join(' | ') : '',
   154→ };
   155→ }));
   156→ break;
   157→ case _TechnologyTab.techStack:
   158→ final text = await ai.generateFepSectionText(
   159→ section: 'Tech Stack',
   160→ context:
   161→ '$ctx\nOutput format (one per line): Name | Description | Status | Cost',
   162→ maxTokens: 700,
   163→ );
   164→ final lines = TextSanitizer.sanitizeAiText(text)
   165→ .split('\n')
   166→ .where((l) => l.trim().isNotEmpty);
   167→ _aiIntegrations
   168→ ..clear()
   169→ ..addAll(lines.map((line) {
   170→ final parts = line.split('|').map((p) => p.trim()).toList();
   171→ return {
   172→ 'name': parts.isNotEmpty ? parts[0] : line.trim(),
   173→ 'description': parts.length > 1 ? parts[1] : '',
   174→ 'status': _normalizeStatus(
   175→ parts.length > 2 ? parts[2] : 'Proposed/Pending'),
   176→ 'cost': parts.length > 3 ? parts[3] : '',
   177→ };
   178→ }));
   179→ break;
   180→ case _TechnologyTab.externalIntegrations:
   181→ final text = await ai.generateFepSectionText(
   182→ section: 'External Integrations',
   183→ context:
   184→ '$ctx\nOutput format (one per line): Name | Description | Status | Connection Type | Complexity | Implementation Cost',
   185→ maxTokens: 700,
   186→ );
   187→ final lines = TextSanitizer.sanitizeAiText(text)
   188→ .split('\n')
   189→ .where((l) => l.trim().isNotEmpty);
   190→ _externalIntegrations
   191→ ..clear()
   192→ ..addAll(lines.map((line) {
   193→ final parts = line.split('|').map((p) => p.trim()).toList();
   194→ return {
   195→ 'name': parts.isNotEmpty ? parts[0] : line.trim(),
   196→ 'description': parts.length > 1 ? parts[1] : '',
   197→ 'status': _normalizeStatus(
   198→ parts.length > 2 ? parts[2] : 'Proposed/Pending'),
   199→ 'connectionType': parts.length > 3 ? parts[3] : 'API',
   200→ 'complexity': parts.length > 4 ? parts[4] : 'Medium',
   201→ 'implementationCost': parts.length > 5 ? parts[5] : '',
   202→ };
   203→ }));
   204→ break;
   205→ case _TechnologyTab.definitions:
   206→ final text = await ai.generateFepSectionText(
   207→ section: 'Technology Definitions',
   208→ context: '$ctx\nOutput format (one per line): Term - Definition',
   209→ maxTokens: 700,
   210→ );
   211→ final lines = TextSanitizer.sanitizeAiText(text)
   212→ .split('\n')
   213→ .where((l) => l.trim().isNotEmpty);
   214→ _definitions
   215→ ..clear()
   216→ ..addAll(lines.map((line) {
   217→ final parts = line.split('-').map((p) => p.trim()).toList();
   218→ return {
   219→ 'term': parts.isNotEmpty ? parts[0] : line.trim(),
   220→ 'definition':
   221→ parts.length > 1 ? parts.sublist(1).join(' - ') : '',
   222→ };
   223→ }));
   224→ break;
   225→ case _TechnologyTab.aiRecommendations:
   226→ final text = await ai.generateFepSectionText(

   230→ maxTokens: 900,

   820→ (double oneTime, double annual) _computeBudget() {
   821→ double oneTime = 0;
   822→ double annual = 0;
   823→
   824→ void absorb(Map<String, dynamic> item, String key) {
   825→ final value = item[key]?.toString() ?? '';
   826→ if (value.trim().isEmpty) return;
   827→ final amount = _parseAmount(value);
   828→ final normalized = value.toLowerCase();
   829→ if (normalized.contains('/month') || normalized.contains('monthly')) {
   830→ annual += amount * 12;
   831→ } else if (normalized.contains('/year') ||
   832→ normalized.contains('annual')) {
   833→ annual += amount;
   834→ } else {
   835→ oneTime += amount;
   836→ }
   837→ }
   838→
   839→ for (final item in _inventory) {
   840→ absorb(item, 'cost');
   841→ }
   842→ for (final item in _aiIntegrations) {
   843→ absorb(item, 'cost');
   844→ }
   845→ for (final item in _externalIntegrations) {
   846→ absorb(item, 'implementationCost');
   847→ }
   848→
   849→ return (oneTime, annual);
   850→ }

  1006→ double? parseAmount(dynamic raw) {
  1007→ final token = raw?.toString() ?? '';
  1008→ final cleaned = token.replaceAll(RegExp(r'[^0-9.]'), '');
  1009→ if (cleaned.isEmpty) return null;
  1010→ return double.tryParse(cleaned);

  1013→ final existingDescriptions = estimate.lines

  1018→ void addCandidate({
  1019→ required String description,
  1020→ required String subCategory,
  1021→ required dynamic amount,
  1022→ }) {

  1026→ additions.add(
  1027→ CostLine(
  1028→ id: '${DateTime.now().microsecondsSinceEpoch}-${additions.length}',
  1029→ category: CostCategory.software,

  1031→ description: description,
  1032→ total: total,
  1033→ basisSource: CostSourceType.kazAI,
  1034→ basisReference: 'Technology Planning',

  1036→ inSchedule: false,

  1041→

  1044→ if (name.isEmpty) continue;
  1045→ addCandidate(
  1046→ description: 'Technology Inventory: $name',
  1047→ subCategory: item['category']?.toString().trim().isNotEmpty == true
  1048→ ? item['category'].toString().trim()
  1049→ : 'Technology Inventory',
  1050→ amount: item['cost'],
  1051→ );
