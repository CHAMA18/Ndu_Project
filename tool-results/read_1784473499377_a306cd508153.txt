   290→ Future<void> _scrollToSectionStart(String id) async {
   291→ final firstContext = _sectionKeys[id]?.currentContext;
   292→ if (firstContext == null || !mounted) return;
   293→ await Future<void>.delayed(const Duration(milliseconds: 40));
   294→ final targetContext = _sectionKeys[id]?.currentContext;
   295→ if (targetContext == null || !mounted || !targetContext.mounted) return;
   296→ await Scrollable.ensureVisible(
   297→ targetContext,
   298→ duration: const Duration(milliseconds: 320),
   299→ curve: Curves.easeInOut,
   300→ alignment: 0.0,
   301→ );
   302→ // Re-run once after tile animation to keep the section header at the top.
   303→ await Future<void>.delayed(const Duration(milliseconds: 260));
   304→ if (!mounted) return;
   305→ final settleContext = _sectionKeys[id]?.currentContext;
   306→ if (settleContext == null || !settleContext.mounted) return;
   307→ await Scrollable.ensureVisible(
   308→ settleContext,

   315→ void _hydrateGuidedSectionState(ProjectDataModel data) {
   316→ final progress = <String, _SectionProgressState>{
   317→ for (final section in _sectionOrder)
   318→ section.id: _SectionProgressState.pending,
   319→ };
   320→ final raw = data.planningNotes[_kSectionProgressNotesKey];
   321→ if (raw != null && raw.trim().isNotEmpty) {
   322→ try {
   323→ final decoded = jsonDecode(raw);
   324→ if (decoded is Map<String, dynamic>) {
   325→ decoded.forEach((key, value) {
   326→ if (!progress.containsKey(key)) return;
   327→ progress[key] = _parseProgressState(value?.toString());
   328→ });
   329→ }
   330→ } catch (e) {
   331→ // Keep defaults if progress payload is malformed.
   332→ }
   333→ }
   334→
   335→ _sectionProgress = progress;
   336→ final requestedSectionId = widget.initialSectionId;
   337→ _activeSectionId =
   338→     _sectionOrder.any((section) => section.id == requestedSectionId)
   339→         ? requestedSectionId!
   340→         : _sectionOrder
   341→             .firstWhere(
   342→               (section) => !_isSectionResolved(section.id),
   343→               orElse: () => _sectionOrder.first,
   344→             )
   345→             .id;
   346→ _sectionExpanded = {
   347→ for (final section in _sectionOrder)
   348→ section.id: section.id == _activeSectionId,
   349→ };
   350→ _sectionTileVersion = {
   351→ for (final section in _sectionOrder) section.id: 0,
   352→ };
   353→ }
   354→
   355→ _SectionProgressState _parseProgressState(String? raw) {
   356→ switch ((raw ?? '').trim().toLowerCase()) {
   357→ case 'complete':
   358→ return _SectionProgressState.complete;
   359→ case 'notapplicable':
   360→ case 'not_applicable':
   361→ return _SectionProgressState.notApplicable;
   362→ default:
   363→ return _SectionProgressState.pending;
   364→ }
   365→ }
   366→
   367→ String _encodeSectionProgress() {
   368→ final payload = <String, String>{
   369→ for (final section in _sectionOrder)
   370→ section.id: _sectionProgress[section.id]!.name,
   371→ };
   372→ return jsonEncode(payload);
   373→ }
   374→
   375→ bool _isSectionResolved(String sectionId) =>
   376→ _sectionProgress[sectionId] != _SectionProgressState.pending;
   377→
   378→ bool _canOpenSection(String sectionId) {
   379→ return _sectionOrder.any((section) => section.id == sectionId);
   380→ }
   381→
   382→ String? _firstBlockingSectionLabel(String sectionId) {
   383→ final targetIndex =
   384→ _sectionOrder.indexWhere((section) => section.id == sectionId);
   385→ if (targetIndex <= 0) return null;
   386→ for (var i = 0; i < targetIndex; i++) {
   387→ final id = _sectionOrder[i].id;
   388→ if (!_isSectionResolved(id)) {
   389→ return _sectionOrder[i].label;
   390→ }
   391→ }
   392→ return null;
   393→ }

   404→ Future<void> _activateSection(String sectionId) async {
   405→ if (!_canOpenSection(sectionId)) {
   406→ _showLockedSectionFeedback(sectionId);
   407→ setState(() {
   408→ // Collapse the section the user tried to open (it was auto-expanded
   409→ // by the ExpansionTile tap before this callback fired).
   410→ _sectionExpanded[sectionId] = false;
   411→ });
   412→ return;
   413→ }
   414→ setState(() {
   415→ // Collapse the previously active section, expand the new one.
   416→ // Modifying the existing map in-place avoids changing the ValueKey
   417→ // for sections whose expanded state didn't change, preventing
   418→ // unnecessary subtree recreation.
   419→ for (final section in _sectionOrder) {
   420→ final shouldExpand = section.id == sectionId;
   421→ if (_sectionExpanded[section.id] != shouldExpand) {
   422→ _sectionExpanded[section.id] = shouldExpand;
   423→ // Bump tile version so the ExpansionTile animates correctly.
   424→ _sectionTileVersion[section.id] =
   425→ (_sectionTileVersion[section.id] ?? 0) + 1;
   426→ }
   427→ }
   428→ _activeSectionId = sectionId;
   429→ });
   430→ await _scrollToSectionStart(sectionId);
   431→ }

   433→ Future<void> _onSectionExpansionChanged(
   434→ String sectionId, bool expanded) async {
   435→ if (!expanded) {
   436→ setState(() => _sectionExpanded[sectionId] = false);

   442→ _sectionExpanded[sectionId] = false;
   443→ _sectionTileVersion[sectionId] =
   444→ (_sectionTileVersion[sectionId] ?? 0) + 1;
   445→ });
   446→ return;
   447→ }
   448→ await _activateSection(sectionId);
   449→ }

   451→ Future<void> _setSectionProgress({
   452→ required String sectionId,
   453→ required _SectionProgressState state,
   454→ }) async {
   455→ final current =
   456→ _sectionProgress[sectionId] ?? _SectionProgressState.pending;
   457→ if (current == state) return;
   458→ if (state == _SectionProgressState.complete &&
   459→ (sectionId == 'requirements' ||
   460→ sectionId == 'design_specifications_workspace')) {
   461→ final requirementOptions =
   462→ _requirementAttachmentOptions(ProjectDataHelper.getData(context));
   463→ final missing = _unlinkedRequirements(requirementOptions);
   464→ if (missing.isNotEmpty) {
   465→ final preview = missing.take(3).map((item) => item.label).join(', ');
   466→ _showToast(
   467→ 'Link specifications to all requirements first. '
   468→ '${missing.length} requirement(s) still unlinked'
   469→ '${preview.isEmpty ? '' : ': $preview'}',
   470→ );
   471→ return;
   472→ }
   473→ }
   474→ if (state != _SectionProgressState.pending) {
   475→ final confirmed = await _confirmStatusChange(state);
   476→ if (!confirmed) return;
   477→ }
   478→ setState(() {
   479→ _sectionProgress[sectionId] = state;
   480→ });
   481→ _queueSave();
   482→ }

   601→ Future<void> _saveDocument() async {
   602→ if (!mounted || _saving) return;
   603→ _updateCoreFields();
   604→ _document.touch();
   605→ _saving = true;
   606→ _saveIndicatorNotifier.value = _SaveIndicatorState(
   607→ saving: _saving,
   608→ pending: _pendingSave,
   609→ lastSavedAt: _lastSavedAt,
   610→ );
   611→ final data = ProjectDataHelper.getData(context);
   612→ final notesPatch = {
   613→ ...data.planningNotes,
