   289→ Future<void> _scrollToSectionStart(String id) async {
   290→ final firstContext = _sectionKeys[id]?.currentContext;
   291→ if (firstContext == null || !mounted) return;
   292→ await Future<void>.delayed(const Duration(milliseconds: 40));
   293→ final targetContext = _sectionKeys[id]?.currentContext;
   294→ if (targetContext == null || !mounted || !targetContext.mounted) return;
   295→ await Scrollable.ensureVisible(
   296→ targetContext,
   297→ duration: const Duration(milliseconds: 320),
   298→ curve: Curves.easeInOut,
   299→ alignment: 0.0,
   300→ );
   301→ // Re-run once after tile animation to keep the section header at the top.
   302→ await Future<void>.delayed(const Duration(milliseconds: 260));
   303→ if (!mounted) return;
   304→ final settleContext = _sectionKeys[id]?.currentContext;

   314→ void _hydrateGuidedSectionState(ProjectDataModel data) {
   315→ final progress = <String, _SectionProgressState>{
   316→ for (final section in _sectionOrder)
   317→ section.id: _SectionProgressState.pending,
   318→ };
   319→ final raw = data.planningNotes[_kSectionProgressNotesKey];
   320→ if (raw != null && raw.trim().isNotEmpty) {
   321→ try {
   322→ final decoded = jsonDecode(raw);
   323→ if (decoded is Map<String, dynamic>) {
   324→ decoded.forEach((key, value) {
   325→ if (!progress.containsKey(key)) return;
   326→ progress[key] = _parseProgressState(value?.toString());
   327→ });
   328→ }
   329→ } catch (e) {
   330→ // Keep defaults if progress payload is malformed.
   331→ }
   332→ }
   333→
   334→ _sectionProgress = progress;
   335→ final requestedSectionId = widget.initialSectionId;
   336→ _activeSectionId =
   337→     _sectionOrder.any((section) => section.id == requestedSectionId)
   338→         ? requestedSectionId!
   339→         : _sectionOrder
   340→             .firstWhere(
   341→               (section) => !_isSectionResolved(section.id),
   342→               orElse: () => _sectionOrder.first,
   343→             )
   344→             .id;
   345→ _sectionExpanded = {
   346→ for (final section in _sectionOrder)
   347→ section.id: section.id == _activeSectionId,
   348→ };
   349→ _sectionTileVersion = {
   350→ for (final section in _sectionOrder) section.id: 0,
   351→ };
   352→ }
   353→
   354→ _SectionProgressState _parseProgressState(String? raw) {
   355→ switch ((raw ?? '').trim().toLowerCase()) {
   356→ case 'complete':
   357→ return _SectionProgressState.complete;
   358→ case 'notapplicable':
   359→ case 'not_applicable':
   360→ return _SectionProgressState.notApplicable;
   361→ default:
   362→ return _SectionProgressState.pending;
   363→ }
   364→ }
   365→
   366→ String _encodeSectionProgress() {
   367→ final payload = <String, String>{
   368→ for (final section in _sectionOrder)
   369→ section.id: _sectionProgress[section.id]!.name,
   370→ };
   371→ return jsonEncode(payload);
   372→ }
   373→
   374→ bool _isSectionResolved(String sectionId) =>
   375→ _sectionProgress[sectionId] != _SectionProgressState.pending;
   376→
   377→ bool _canOpenSection(String sectionId) {
   378→ return _sectionOrder.any((section) => section.id == sectionId);
   379→ }
   380→
   381→ String? _firstBlockingSectionLabel(String sectionId) {
   382→ final targetIndex =
   383→ _sectionOrder.indexWhere((section) => section.id == sectionId);
   384→ if (targetIndex <= 0) return null;
   385→ for (var i = 0; i < targetIndex; i++) {
   386→ final id = _sectionOrder[i].id;
   387→ if (!_isSectionResolved(id)) {
   388→ return _sectionOrder[i].label;
   389→ }
   390→ }
   391→ return null;
   392→ }

   403→ Future<void> _activateSection(String sectionId) async {
   404→ if (!_canOpenSection(sectionId)) {
   405→ _showLockedSectionFeedback(sectionId);
   406→ setState(() {
   407→ // Collapse the section the user tried to open (it was auto-expanded
   408→ // by the ExpansionTile tap before this callback fired).
   409→ _sectionExpanded[sectionId] = false;
   410→ });
   411→ return;
   412→ }
   413→ setState(() {
   414→ // Collapse the previously active section, expand the new one.
   415→ // Modifying the existing map in-place avoids changing the ValueKey
   416→ // for sections whose expanded state didn't change, preventing
   417→ // unnecessary subtree recreation.
   418→ for (final section in _sectionOrder) {
   419→ final shouldExpand = section.id == sectionId;
   420→ if (_sectionExpanded[section.id] != shouldExpand) {
   421→ _sectionExpanded[section.id] = shouldExpand;
   422→ // Bump tile version so the ExpansionTile animates correctly.
   423→ _sectionTileVersion[section.id] =
   424→ (_sectionTileVersion[section.id] ?? 0) + 1;
   425→ }
   426→ }
   427→ _activeSectionId = sectionId;
   428→ });
   429→ await _scrollToSectionStart(sectionId);
   430→ }

   432→ Future<void> _onSectionExpansionChanged(
   433→ String sectionId, bool expanded) async {
   434→ if (!expanded) {
   435→ setState(() => _sectionExpanded[sectionId] = false);

   443→ (_sectionTileVersion[sectionId] ?? 0) + 1;

   445→ return;
   446→ }
   447→ await _activateSection(sectionId);
   448→ }

   450→ Future<void> _setSectionProgress({
   451→ required String sectionId,
   452→ required _SectionProgressState state,
   453→ }) async {
   454→ final current =
   455→ _sectionProgress[sectionId] ?? _SectionProgressState.pending;
   456→ if (current == state) return;
   457→ if (state == _SectionProgressState.complete &&
   458→ (sectionId == 'requirements' ||
   459→ sectionId == 'design_specifications_workspace')) {
   460→ final requirementOptions =
   461→ _requirementAttachmentOptions(ProjectDataHelper.getData(context));
   462→ final missing = _unlinkedRequirements(requirementOptions);
   463→ if (missing.isNotEmpty) {
   464→ final preview = missing.take(3).map((item) => item.label).join(', ');
   465→ _showToast(
   466→ 'Link specifications to all requirements first. '
   467→ '${missing.length} requirement(s) still unlinked'
   468→ '${preview.isEmpty ? '' : ': $preview'}',
   469→ );
   470→ return;
   471→ }
   472→ }
   473→ if (state != _SectionProgressState.pending) {
   474→ final confirmed = await _confirmStatusChange(state);
   475→ if (!confirmed) return;
   476→ }
   477→ setState(() {
   478→ _sectionProgress[sectionId] = state;
   479→ });
   480→ _queueSave();
   481→ }

   600→ Future<void> _saveDocument() async {
   601→ if (!mounted || _saving) return;
   602→ _updateCoreFields();
   603→ _document.touch();
   604→ _saving = true;
   605→ _saveIndicatorNotifier.value = _SaveIndicatorState(
   606→ saving: _saving,
   607→ pending: _pendingSave,
   608→ lastSavedAt: _lastSavedAt,
   609→ );
   610→ final data = ProjectDataHelper.getData(context);
   611→ final notesPatch = {
   612→ ...data.planningNotes,
   613→ ..._document.toPlanningNotesPatch(),
   614→ _kSectionProgressNotesKey: _encodeSectionProgress(),
