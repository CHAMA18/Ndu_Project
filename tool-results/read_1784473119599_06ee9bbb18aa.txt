   105→  Future<void> _autoPopulateFromProjectData(BuildContext context) async {
   106→    final data = ProjectDataHelper.getData(context);
   107→    if (data.planningNotes['interface_auto_populated'] == 'true') return;
   108→    if (data.interfaceEntries.isNotEmpty) return;
   109→
   110→    final entries = <InterfaceEntry>[];
   111→    final now = DateTime.now().toIso8601String();
   112→    final logEntries = List<InterfaceChangeLogEntry>.from(data.interfaceChangeLog);
   113→
   114→    // From Vendors
   115→    for (final v in data.vendors) {
   116→      final name = v.name.trim();
   117→      if (name.isEmpty) continue;
   118→      entries.add(InterfaceEntry(
   119→        boundary: name,
   120→        interfaceType: 'Contractual',
   121→        partyA: 'Project Team',
   122→        partyB: name,
   123→        criticality: 'Major',
   124→        priority: 'Medium',
   125→        status: 'Pending',
   126→        notes: 'Vendor interface - ${v.equipmentOrService.trim()}',
   127→      ));
   128→    }
   129→
   130→    // From Contractors
   131→    for (final c in data.contractors) {
   132→      final name = c.name.trim();
   133→      if (name.isEmpty) continue;
   134→      entries.add(InterfaceEntry(
   135→        boundary: name,
   136→        interfaceType: 'Contractual',
   137→        partyA: 'Project Team',
   138→        partyB: name,
   139→        criticality: 'Major',
   140→        priority: 'Medium',
   141→        status: 'Pending',
   142→        notes: 'Contractor interface - ${c.service.trim()}',
   143→      ));
   144→    }
   145→
   146→    // From External Integrations (Technology Planning)
   147→    for (final ext in data.externalIntegrations) {
   148→      final name = ext['name']?.toString().trim() ?? '';
   149→      if (name.isEmpty) continue;
   150→      entries.add(InterfaceEntry(
   151→        boundary: name,
   152→        interfaceType: 'Technical',
   153→        partyA: 'Project Team',
   154→        partyB: name,
   155→        criticality: 'Major',
   156→        priority: 'Medium',
   157→        status: 'Pending',
   158→        owner: ext['description']?.toString() ?? '',
   159→        notes: 'Technical interface from Technology Planning',
   160→        id: '',
   161→      ));
   162→    }
   163→
   164→    if (entries.isEmpty) return;
   165→
   166→    for (final entry in entries) {
   167→      logEntries.add(InterfaceChangeLogEntry(
   168→        interfaceId: entry.id,
   169→        interfaceName: entry.boundary,
   170→        action: 'Auto-Populated',
   171→        newValue: entry.boundary,
   172→        changedAt: now,
   173→      ));
   174→    }
   175→
   176→    await ProjectDataHelper.updateAndSave(
   177→      context: context,
   178→      checkpoint: 'interface_management',
   179→      dataUpdater: (d) => d.copyWith(
   180→        planningNotes: {
   181→          ...d.planningNotes,
   182→          'interface_auto_populated': 'true',
   183→        },
   184→        interfaceEntries: [...d.interfaceEntries, ...entries],
   185→        interfaceChangeLog: [...d.interfaceChangeLog, ...logEntries],
   186→      ),
   187→      showSnackbar: true,
   188→    );
   189→
   190→    if (mounted) {
   191→      ScaffoldMessenger.of(context).showSnackBar(
   192→        SnackBar(
   193→          content: Text('Auto-populated ${entries.length} interface(s) from project data'),
   194→        ),
   195→      );
   196→    }
   197→    unawaited(ActivityLogService.instance.logCheckpointActivity(
   198→      projectId: data.projectId ?? '',
   199→      checkpoint: 'interface_management',
   200→      action: 'Interface Auto-Populated',
   201→      details: {'count': entries.length},
   202→    ));
   203→
   204→    // After basic auto-populate, kick off AI generation for additional
   205→    // cross-scope interfaces (vendors, contractors, other orgs, program/portfolio)
   206→    if (mounted) {
   207→      unawaited(_aiPopulateInterfaces(context, data));
   208→    }

   211→  /// AI-enhanced interface population. Calls generateInterfaceEntries to
   212→  /// identify cross-scope interfaces with vendors, contractors, other
   213→  /// companies/organizations, and program/portfolio projects. De-duplicates
   214→  /// against existing entries by boundary name.
   215→  /// 
   216→  /// Enhanced to include comprehensive project context:
   217→  /// - Stakeholder register data
   218→  /// - Vendor/contractor information from Procurement
   219→  /// - Existing interfaces already identified
   220→  /// - Company/organization info where available
   221→  Future<void> _aiPopulateInterfaces(
   222→      BuildContext context, ProjectDataModel data) async {
   223→    if (data.planningNotes['interface_ai_populated'] == 'true') return;
   224→
   225→    final additionalInterfaces =
   226→        data.planningNotes['additional_interfaces'] ?? 'Not Applicable';
   227→
   228→    // Build a comprehensive program/portfolio context including:
   229→    // - Stakeholder register data
   230→    // - Vendor/contractor info
   231→    // - Organization details
   232→    // - Existing interfaces
   233→    final programPortfolioContext = <String>[];
   234→    
   235→    try {
   236→      // 1. Stakeholder Management Plan summary from planning notes
   237→      final stakeholderPlan =
   238→          data.planningNotes['stakeholder_management_plan'] ??
   239→          data.planningNotes['stakeholder_engagement_plan'] ??
   240→          '';
   241→      if (stakeholderPlan.isNotEmpty) {
   242→        programPortfolioContext
   243→            .add('Stakeholder Management Plan: $stakeholderPlan');
   244→      }
   245→      
   246→      // 2. Stakeholder Register Entries - extract key stakeholders for interface analysis
   247→      if (data.stakeholderEntries.isNotEmpty) {
   248→        final stakeholderSummary = data.stakeholderEntries
   249→            .where((s) => (s.name ?? '').trim().isNotEmpty)
   250→            .take(15)
   251→            .map((s) => '- ${s.name} (${s.role ?? "Unknown Role"}): ${s.influence ?? ""} influence, ${s.interest ?? ""} interest')
   252→            .join('\n');
   253→        if (stakeholderSummary.isNotEmpty) {
   254→          programPortfolioContext.add('Key Stakeholders:\n$stakeholderSummary');
   255→        }
   256→      }
   257→      
   258→      // 3. Vendor Information from Procurement
   259→      if (data.vendors.isNotEmpty) {
   260→        final vendorSummary = data.vendors
   261→            .where((v) => v.name.trim().isNotEmpty)
   262→            .map((v) => '- ${v.name}: ${v.equipmentOrService} (${v.status})')
   263→            .join('\n');
   264→        if (vendorSummary.isNotEmpty) {
   265→          programPortfolioContext.add('Procurement Vendors:\n$vendorSummary');
   266→        }
   267→      }
   268→      
   269→      // 4. Contractor Information from Procurement
   270→      if (data.contractors.isNotEmpty) {
   271→        final contractorSummary = data.contractors
   272→            .where((c) => c.name.trim().isNotEmpty)
   273→            .map((c) => '- ${c.name}: ${c.service} (${c.status})')
   274→            .join('\n');
   275→        if (contractorSummary.isNotEmpty) {
   276→          programPortfolioContext.add('Contractors:\n$contractorSummary');
   277→        }
   278→      }
   279→      
   280→      // 5. Existing interfaces already identified (for de-duplication context)
   281→      if (data.interfaceEntries.isNotEmpty) {
   282→        final existingIfaces = data.interfaceEntries
   283→            .where((e) => e.boundary.trim().isNotEmpty)
   284→            .take(10)
   285→            .map((e) => '- ${e.boundary} [${e.interfaceType}] ${e.partyA} ↔ ${e.partyB}')
   286→            .join('\n');
   287→        if (existingIfaces.isNotEmpty) {
   288→          programPortfolioContext.add('Existing Interfaces Already Identified:\n$existingIfaces');
   289→        }
   290→      }
   291→      
   292→      // 6. External integrations from Technology Planning
   293→      if (data.externalIntegrations.isNotEmpty) {
   294→        final extIntegrations = data.externalIntegrations
   295→            .where((e) => (e['name']?.toString().trim() ?? '').isNotEmpty)
   296→            .map((e) => '- ${e["name"]}: ${e["description"] ?? ""} (${e["connectionType"] ?? "API"})')
   297→            .join('\n');
   298→        if (extIntegrations.isNotEmpty) {
   299→          programPortfolioContext.add('Technology External Integrations:\n$extIntegrations');
   300→        }
   301→      }
   302→      
   303→      // 7. Project site/location info if available
   304→      final siteInfo = data.planningNotes['project_site'] ?? 
   305→          data.planningNotes['site_location'] ?? 
   306→          data.planningNotes['location'] ?? '';
   307→      if (siteInfo.isNotEmpty) {
   308→        programPortfolioContext.add('Project Site/Location: $siteInfo');
   309→      }
   310→      
   311→    } catch (_) {
   312→      debugPrint('Error building program/portfolio context for AI interface generation');
   313→    }
   314→
   315→    final projectContext =
   316→        ProjectDataHelper.buildFepContext(data, sectionLabel: 'Interface Management');
   317→    if (projectContext.trim().isEmpty) return;
   318→
   319→    List<Map<String, dynamic>> aiEntries = [];
   320→    try {
   321→      aiEntries = await OpenAiServiceSecure().generateInterfaceEntries(
   322→        context: projectContext,
   323→        additionalInterfaces: additionalInterfaces,
   324→        programPortfolioContext: programPortfolioContext.join('\n\n'),
   325→        maxItems: 10,
   326→      );
   327→    } catch (e) {
   328→      debugPrint('AI interface generation failed: $e');
   329→      return;
   330→    }
   331→
   332→    if (!mounted || aiEntries.isEmpty) {
   333→      // Mark as attempted so we don't retry on every rebuild
   334→      await ProjectDataHelper.updateAndSave(
   335→        context: context,
   336→        checkpoint: 'interface_management',
   337→        showSnackbar: false,
   338→        dataUpdater: (d) => d.copyWith(
   339→          planningNotes: {
   340→            ...d.planningNotes,
   341→            'interface_ai_populated': 'true',
   342→          },
   343→        ),
   344→      );
   345→      return;
   346→    }
   347→
   348→    // De-duplicate against existing entries by boundary name (case-insensitive)
   349→    final existingBoundaries = data.interfaceEntries
   350→        .map((e) => e.boundary.trim().toLowerCase())
   351→        .where((s) => s.isNotEmpty)
   352→        .toSet();
   353→
   354→    final newEntries = <InterfaceEntry>[];
   355→    final logEntries = <InterfaceChangeLogEntry>[];
   356→    final now = DateTime.now().toIso8601String();
   357→
   358→    for (final m in aiEntries) {
   359→      final boundary = (m['boundary'] ?? '').toString().trim();
   360→      if (boundary.isEmpty) continue;
   361→      if (existingBoundaries.contains(boundary.toLowerCase())) continue;
   362→      newEntries.add(InterfaceEntry(
   363→        boundary: boundary,
   364→        interfaceType: (m['interfaceType'] ?? 'Organizational').toString(),
   365→        partyA: (m['partyA'] ?? 'Project Team').toString(),
   366→        partyB: (m['partyB'] ?? '').toString(),
   367→        criticality: (m['criticality'] ?? 'Major').toString(),
   368→        priority: (m['priority'] ?? 'Medium').toString(),
   369→        status: (m['status'] ?? 'Pending').toString(),
   370→        owner: (m['owner'] ?? '').toString(),
   371→        notes: (m['notes'] ?? '').toString(),
   372→      ));
   373→      existingBoundaries.add(boundary.toLowerCase());
   374→      logEntries.add(InterfaceChangeLogEntry(
   375→        interfaceId: '',
   376→        interfaceName: boundary,
   377→        action: 'AI-Generated',
   378→        newValue: boundary,
   379→        changedAt: now,
   380→      ));
   381→    }
   382→
   383→    if (newEntries.isEmpty) {
   384→      await ProjectDataHelper.updateAndSave(
   385→        context: context,
   386→        checkpoint: 'interface_management',
   387→        showSnackbar: false,
   388→        dataUpdater: (d) => d.copyWith(
   389→          planningNotes: {
   390→            ...d.planningNotes,
   391→            'interface_ai_populated': 'true',
   392→          },
   393→        ),
   394→      );
   395→      return;
   396→    }
   397→
   398→    await ProjectDataHelper.updateAndSave(
   399→      context: context,
   400→      checkpoint: 'interface_management',
   401→      showSnackbar: false,
   402→      dataUpdater: (d) => d.copyWith(
   403→        planningNotes: {
   404→          ...d.planningNotes,
   405→          'interface_ai_populated': 'true',
   406→        },
   407→        interfaceEntries: [...d.interfaceEntries, ...newEntries],
   408→        interfaceChangeLog: [...d.interfaceChangeLog, ...logEntries],
   409→      ),
   410→    );
   411→
   412→    if (!mounted) return;
   413→    ScaffoldMessenger.of(context).showSnackBar(
   414→      SnackBar(
   415→        content: Text(
   416→            'KAZ AI identified ${newEntries.length} additional cross-scope interface(s).'),
   417→        duration: const Duration(seconds: 4),
   418→      ),
   419→    );
   420→  }

   721→ Future<void> _generateAiPlan() async {
   722→   if (_isAiGenerating) return;
   723→   setState(() => _isAiGenerating = true);
   724→   try {
   725→     final data = ProjectDataHelper.getData(context);
   726→     final additionalInterfaces =
   727→         data.planningNotes['additional_interfaces'] ?? 'Not Applicable';
   728→     
   729→     // Build enhanced stakeholder plan context

   731→     final baseStakeholderPlan =
   732→         data.planningNotes['stakeholder_management_plan'] ??
   733→         data.planningNotes['stakeholder_engagement_plan'] ??
   734→         '';
   735→     if (baseStakeholderPlan.isNotEmpty) {
   736→       stakeholderPlanBuffer.writeln(baseStakeholderPlan);
   737→       stakeholderPlanBuffer.writeln();
   738→     }
   739→     
   740→     // Add vendor/contractor context for richer AI plan generation
   741→     if (data.vendors.isNotEmpty) {
   742→       stakeholderPlanBuffer.writeln('Procurement Vendors:');
   743→       for (final v in data.vendors.take(8)) {
   744→         if (v.name.trim().isNotEmpty) {
   745→           stakeholderPlanBuffer.writeln('- ${v.name}: ${v.equipmentOrService}');
   746→         }
   747→       }
   748→       stakeholderPlanBuffer.writeln();
   749→     }
   750→     
   751→     if (data.contractors.isNotEmpty) {
   752→       stakeholderPlanBuffer.writeln('Contractors:');
   753→       for (final c in data.contractors.take(6)) {
   754→         if (c.name.trim().isNotEmpty) {
   755→           stakeholderPlanBuffer.writeln('- ${c.name}: ${c.service}');
   756→         }
   757→       }
   758→       stakeholderPlanBuffer.writeln();
   759→     }
   760→     
   761→     // Add existing interface summary for context
   762→     if (data.interfaceEntries.isNotEmpty) {
   763→       stakeholderPlanBuffer.writeln('Current Interface Register Summary (${data.interfaceEntries.length} entries):');
   764→       final criticalIfaces = data.interfaceEntries.where((e) => e.criticality.toLowerCase() == 'critical').toList();
   765→       final majorIfaces = data.interfaceEntries.where((e) => e.criticality.toLowerCase() == 'major').toList();
   766→       if (criticalIfaces.isNotEmpty) {
   767→         stakeholderPlanBuffer.writeln('Critical Interfaces: ${criticalIfaces.map((e) => e.boundary).join(", ")}');
   768→       }
   769→       if (majorIfaces.isNotEmpty) {
   770→         stakeholderPlanBuffer.writeln('Major Interfaces: ${majorIfaces.map((e) => e.boundary).join(", ")}');
   771→       }
   772→       stakeholderPlanBuffer.writeln();

   775→     final enhancedStakeholderPlan = stakeholderPlanBuffer.toString().trim();
   776→     
   777→     final projectContext = ProjectDataHelper.buildFepContext(
   778→         data, sectionLabel: 'Interface Management');

   780→       setState(() => _isAiGenerating = false);

   784→       context: projectContext,
   785→       additionalInterfaces: additionalInterfaces,
   786→       stakeholderPlanSummary: enhancedStakeholderPlan,
   787→     );
   788→     if (!mounted) return;
   789→     if (plan.trim().isNotEmpty) {
   790→       setState(() => _initialText = plan.trim());
   791→       await ProjectDataHelper.updateAndSave(
   792→         context: context,
   793→         checkpoint: 'interface_management',
   794→         showSnackbar: false,
   795→         dataUpdater: (d) => d.copyWith(
   796→           planningNotes: {
   797→             ...d.planningNotes,
   798→             _noteKey: plan.trim(),
   799→             'interface_plan_ai_generated': 'true',
   800→           },
   801→         ),
   802→       );
   803→       if (mounted) {
   804→         setState(() => _lastSavedAt = DateTime.now());
   805→       }
   806→     }

   808→     debugPrint('AI Interface Management Plan generation failed: $e');
   809→   } finally {
   810→     if (mounted) setState(() => _isAiGenerating = false);
   811→   }
   812→ }

  1595→void _syncInterfaceRiskToGlobal({
  1596→  required BuildContext context,
  1597→  required InterfaceEntry entry,
  1598→  InterfaceEntry? oldEntry,
  1599→}) {
  1600→  final data = ProjectDataHelper.getData(context);
  1601→  var riskItems = List<ExecutionRiskItem>.from(data.executionRiskItems);
  1602→  final existingIdx = riskItems.indexWhere((r) => r.id == 'iface_${entry.id}');
  1603→  final hasNewRisk = entry.risk.trim().isNotEmpty;
  1604→
  1605→  if (!hasNewRisk) {
  1606→    if (existingIdx != -1) {
  1607→      riskItems.removeAt(existingIdx);
  1608→    }
  1609→  } else {
  1610→    final newItem = _interfaceEntryToRiskItem(entry);
  1611→    if (existingIdx != -1) {
  1612→      riskItems[existingIdx] = newItem;
  1613→    } else {
  1614→      riskItems.add(newItem);
  1615→    }
  1616→  }
  1617→
  1618→  if (existingIdx != -1 || hasNewRisk) {
  1619→    ProjectDataHelper.updateAndSave(
  1620→      context: context,
  1621→      checkpoint: 'interface_management',
  1622→      dataUpdater: (d) => d.copyWith(executionRiskItems: riskItems),
  1623→      showSnackbar: false,
  1624→    );
  1625→  }
  1626→
  1627→  // Also sync to the central FEP Risk Register (bidirectional)
  1628→  if (hasNewRisk) {
  1629→    unawaited(ProjectDataHelper.upsertRiskToRegister(
  1630→      context: context,
  1631→      sourceSection: 'Interface Management',
  1632→      riskName: 'Interface: ${entry.boundary}',
  1633→      description: entry.notes.isNotEmpty
  1634→          ? entry.notes
  1635→          : 'Risk associated with interface ${entry.boundary} (${entry.partyA} ↔ ${entry.partyB})',
  1636→      category: 'Interface Management',
  1637→      impactLevel: entry.criticality.toLowerCase() == 'critical'
  1638→          ? 'High'
  1639→          : entry.criticality.toLowerCase() == 'major'
  1640→              ? 'Medium'
  1641→              : 'Low',
  1642→      likelihood: 'Medium',
  1643→      mitigationStrategy: entry.risk,
  1644→      discipline: entry.interfaceType,
  1645→      owner: entry.owner,
  1646→      status: entry.status,
  1647→    ));
  1648→  } else {
  1649→    unawaited(ProjectDataHelper.removeRiskFromRegister(
  1650→      context: context,
  1651→      sourceSection: 'Interface Management',
  1652→      riskName: 'Interface: ${entry.boundary}',
  1653→    ));
  1654→  }
  1655→
  1656→  // Log to central activities log
  1657→  unawaited(ProjectDataHelper.logActivityToCentral(
  1658→    context: context,
  1659→    sourceSection: 'Interface Management',
  1660→    title: 'Interface ${entry.boundary} — ${hasNewRisk ? "Risk identified" : "Risk cleared"}',
  1661→    description: 'Interface between ${entry.partyA} and ${entry.partyB}. Risk: ${entry.risk}',
  1662→    phase: 'Planning',
  1663→    discipline: entry.interfaceType,
  1664→    role: entry.owner,
  1665→  ));
  1666→}
