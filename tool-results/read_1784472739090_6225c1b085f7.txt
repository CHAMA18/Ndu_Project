  1000→  factory ProjectDataModel.fromJson(Map<String, dynamic> json) {
  1001→    // Reconstruct goalWorkItems from flattened structure
  1002→    List<List<WorkItem>> reconstructedGoalWorkItems = List.generate(
  1003→      3,
  1004→      (_) => [],
  1005→    );
  1006→    final rawWorkItems = json['goalWorkItems'] as List?;
  1007→
  1008→    if (rawWorkItems != null) {
  1009→      try {
  1010→        // Check if it's the old nested format or new flattened format
  1011→        if (rawWorkItems.isNotEmpty && rawWorkItems.first is List) {
  1012→          // Old nested format (backward compatibility)
  1013→          reconstructedGoalWorkItems = rawWorkItems
  1014→              .map(
  1015→                (items) =>
  1016→                    (items as List).map((i) => WorkItem.fromJson(i)).toList(),
  1017→              )
  1018→              .toList();
  1019→        } else {
  1020→          // New flattened format
  1021→          for (final item in rawWorkItems) {
  1022→            final itemMap = item as Map<String, dynamic>;
  1023→            final goalIndex = itemMap['goalIndex'] as int? ?? 0;
  1024→
  1025→            // Ensure the list is large enough
  1026→            while (reconstructedGoalWorkItems.length <= goalIndex) {
  1027→              reconstructedGoalWorkItems.add([]);
  1028→            }
  1029→
  1030→            reconstructedGoalWorkItems[goalIndex].add(
  1031→              WorkItem.fromJson(itemMap),
  1032→            );
  1033→          }
  1034→        }
  1035→      } catch (e) {
  1036→        debugPrint('⚠️ Error parsing goalWorkItems: $e');
  1037→        reconstructedGoalWorkItems = List.generate(3, (_) => []);
  1038→      }
  1039→    }

  1458→  /// ── P2.5: Compute project-level aggregate EVM from control accounts ──
  1459→  /// Rolls up BAC, EV, AC, and PV from all [controlAccounts], then derives
  1460→  /// CPI, SPI, EAC, ETC, VAC, CV, SV, and TCPI using standard EVM formulas.
  1461→  ///
  1462→  /// Call this after [ControlAccountService.recalculateAll] has updated
  1463→  /// individual control account EVM metrics. This method mutates the aggregate
  1464→  /// fields on this model in-place and updates [evmLastRecalculated].
  1465→  ///
  1466→  /// Returns `this` for chaining convenience.
  1467→  ProjectDataModel computeAggregateEvm() {
  1468→    double aggBac = 0, aggEv = 0, aggAc = 0, aggPv = 0;
  1469→    for (final ca in controlAccounts) {
  1470→      aggBac += ca.budgetAtCompletion;
  1471→      aggEv += ca.earnedValue;
  1472→      aggAc += ca.actualCost;
  1473→      // Sum planned value from period data
  1474→      final now = DateTime.now();
  1475→      final currentKey = '${now.year}-${now.month.toString().padLeft(2, '0')}';
  1476→      for (final entry in ca.plannedValueByPeriod.entries) {
  1477→        if (entry.key.compareTo(currentKey) <= 0) {
  1478→          aggPv += entry.value;
  1479→        }
  1480→      }
  1481→    }
  1482→
  1483→    aggregateBac = aggBac;
  1484→    aggregatePlannedValue = aggPv;
  1485→    aggregateEarnedValue = aggEv;
  1486→    aggregateActualCost = aggAc;
  1487→    aggregateCpi = aggAc > 0 ? aggEv / aggAc : 1.0;
  1488→    aggregateSpi = aggPv > 0 ? aggEv / aggPv : 1.0;
  1489→    aggregateEac = aggregateCpi > 0 ? aggBac / aggregateCpi : aggBac;
  1490→    aggregateEtc = aggregateEac - aggAc;
  1491→    aggregateVac = aggBac - aggregateEac;
  1492→    aggregateCv = aggEv - aggAc;
  1493→    aggregateSv = aggEv - aggPv;
  1494→    aggregateTcpi = (aggBac - aggAc) > 0
  1495→        ? (aggBac - aggEv) / (aggBac - aggAc)
  1496→        : 1.0;
  1497→    evmLastRecalculated = DateTime.now();
  1498→
  1499→    return this;
  1500→  }
