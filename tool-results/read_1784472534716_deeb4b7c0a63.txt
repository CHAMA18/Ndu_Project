   121→    // We'll hydrate from provider in didChangeDependencies.

   135→
   136→    // Check admin status

   139→    });
   140→    ApiKeyManager.initializeApiKey();
   141→    WidgetsBinding.instance.addPostFrameCallback((_) {
   142→      if (!mounted) return;

   144→        _loadExistingData();

   146→        debugPrint('initState error: $e');

   149→      // Only auto-generate if there is NO existing IT data with content

   152→      final existingItData = provider?.projectData.itConsiderationsData;
   153→      final hasExistingData = existingItData != null &&
   154→          existingItData.solutionITData.isNotEmpty &&

   156→              .any((item) => item.coreTechnology.trim().isNotEmpty);
   157→
   158→      // Ensure _techControllers aligns with ALL solutions
   159→      if (mounted) {
   160→        setState(() {
   161→          while (_techControllers.length < _solutions.length) {
   162→            final controller = RichTextEditingController();
   163→            controller.enableAutoBullet();
   164→            _techControllers.add(controller);
   165→          }
   166→        });
   167→      }
   168→
   169→      if (_solutions.isNotEmpty && !hasExistingData) {
   170→        _generateTechnologies();
   171→      }
   172→
   173→      Future.delayed(const Duration(milliseconds: 500), () {
   174→        if (!mounted) return;
   175→        PageHintDialog.showIfNeeded(
   176→          context: context,
   177→          pageId: 'it_considerations',
   178→          title: 'IT Considerations',
   179→          message:
   180→              'List the core technology considerations for each potential solution. Click "Generate Technologies" to get AI suggestions tailored to each solution.',
   181→        );
   182→      });
   183→    });
   184→  }

   199→  void _loadExistingData() {
   200→    try {
   201→      final provider = ProjectDataInherited.read(context);
   202→      final itData = provider.projectData.itConsiderationsData;
   203→
   204→      if (itData == null) return;
   205→
   206→      // Load notes
   207→      if (itData.notes.isNotEmpty) {
   208→        _notesController.text = itData.notes;
   209→      }
   210→
   211→      // Load IT data for each solution
   212→      final maxItems = itData.solutionITData.length;
   213→
   214→      while (_techControllers.length < maxItems) {
   215→        _solutions.add(AiSolutionItem(title: '', description: ''));
   216→        final newController = RichTextEditingController();
   217→        newController.enableAutoBullet(); // Enable auto-bullet for new field
   218→        _techControllers.add(newController);
   219→      }
   220→      for (int i = 0; i < maxItems && i < _techControllers.length; i++) {
   221→        final solutionIT = itData.solutionITData[i];
   222→        if (i < _solutions.length) {
   223→          _solutions[i] = AiSolutionItem(
   224→            title: _cleanSolutionTitle(solutionIT.solutionTitle),
   225→            description: '',
   226→          );
   227→        }
   228→        _techControllers[i].text = solutionIT.coreTechnology;
   229→      }
   230→
   231→      if (mounted) setState(() {});
   232→    } catch (e) {
   233→      debugPrint('Error loading existing IT considerations data: $e');
   234→    }
   235→  }

   237→  Future<void> _generateTechnologies() async {
   238→    if (_isGenerating) return;
   239→    setState(() {
   240→      _isGenerating = true;
   241→      _error = null;
   242→    });
   243→    try {
   244→      final provider = ProjectDataHelper.getProvider(context);
   245→
   246→      // Add current values to history before regenerating
   247→      for (int i = 0;
   248→          i < _solutions.length && i < _techControllers.length;
   249→          i++) {
   250→        final fieldKey = 'it_tech_${_solutions[i].title}_$i';
   251→        provider.addFieldToHistory(fieldKey, _techControllers[i].text,
   252→            isAiGenerated: true);
   253→      }
   254→
   255→      // Get project context for fallback if solutions are empty
   256→      final projectData = provider.projectData;
   257→      final projectName = projectData.projectName;
   258→      final projectDescription = projectData.solutionDescription;
   259→
   260→      // Use solutions if available, otherwise create a placeholder from project name
   261→      final solutionsToUse = _solutions
   262→          .where((s) => s.title.isNotEmpty || s.description.isNotEmpty)
   263→          .toList();
   264→      if (solutionsToUse.isEmpty && projectName.isNotEmpty) {
   265→        solutionsToUse.add(AiSolutionItem(
   266→          title: projectName,
   267→          description: projectDescription,
   268→        ));
   269→        // Ensure we have a controller for this
   270→        if (_techControllers.isEmpty) {
   271→          _techControllers.add(RichTextEditingController());
   272→        }
   273→        if (_solutions.isEmpty) {
   274→          _solutions.addAll(solutionsToUse);
   275→        }
   276→      }
   277→
   278→      if (solutionsToUse.isEmpty) {
   279→        setState(() {
   280→          _error =
   281→              'Please add at least one solution or project name to generate IT considerations.';
   282→          _isGenerating = false;
   283→        });
   284→        return;
   285→      }
   286→
   287→      // Build context notes with project info if available
   288→      String contextNotes = _notesController.text.trim();
   289→      if (contextNotes.isEmpty && projectName.isNotEmpty) {
   290→        contextNotes = 'Project: $projectName';
   291→        if (projectDescription.isNotEmpty) {
   292→          contextNotes += '\nDescription: $projectDescription';
   293→        }
   294→      }
   295→
   296→      final map = await _openAi.generateTechnologiesForSolutions(
   297→        solutionsToUse,
   298→        contextNotes: contextNotes,
   299→      );
   300→
   301→      // Apply generated data to controllers
   302→      for (int i = 0;
   303→          i < solutionsToUse.length && i < _techControllers.length;
   304→          i++) {
   305→        final title = solutionsToUse[i].title;
   306→        final tech = map[title] ?? const <String>[];
   307→        _techControllers[i].text = tech.isEmpty
   308→            ? 'Not applicable based on the current project context. Add your own IT considerations for this solution.'
   309→            : tech.map((e) => '- $e').join('\n');
   310→      }
   311→
   312→      // Auto-save after regeneration
   313→      await provider.saveToFirebase(
   314→          checkpoint: 'it_considerations_regenerated');
   315→
   316→      if (mounted) {
   317→        ScaffoldMessenger.of(context).showSnackBar(
   318→          const SnackBar(
   319→              content: Text('IT considerations regenerated successfully')),
   320→        );
   321→      }
   322→    } catch (e) {
   323→      _error = (e.toString().contains('Failed to fetch') ||
   324→              e.toString().contains('ClientException') ||
   325→              e.toString().contains('XMLHttpRequest') ||
   326→              e.toString().contains('Connection refused'))
   327→          ? 'AI assist is being set up. Please try again later or enter content manually.'
   328→          : e.toString();
   329→      if (mounted) {
   330→        ScaffoldMessenger.of(context).showSnackBar(
   331→          SnackBar(content: Text('Failed to regenerate IT considerations: $e')),
   332→        );
   333→      }
   334→    } finally {
   335→      if (mounted) {
   336→        setState(() => _isGenerating = false);
   337→        // Auto-save after generation
   338→        _saveITConsiderationsData();
   339→      }
   340→    }
   341→  }

  1133→  Future<List<_ItAutoFillPreviewRow>> _buildItAutofillPreview() async {
  1134→    final provider = ProjectDataHelper.getProvider(context);
  1135→    final projectData = provider.projectData;
  1136→    final solutionsToUse = _resolveItSolutionsForAutofill(projectData);
  1137→    if (solutionsToUse.isEmpty) return const <_ItAutoFillPreviewRow>[];
  1138→
  1139→    var contextNotes = _notesController.text.trim();
  1140→    if (contextNotes.isEmpty && projectData.projectName.trim().isNotEmpty) {
  1141→      contextNotes = 'Project: ${projectData.projectName.trim()}';
  1142→      if (projectData.solutionDescription.trim().isNotEmpty) {
  1143→        contextNotes +=
  1144→            '\nDescription: ${projectData.solutionDescription.trim()}';
  1145→      }
  1146→    }
  1147→
  1148→    final generated = await _openAi.generateTechnologiesForSolutions(
  1149→      solutionsToUse,
  1150→      contextNotes: contextNotes,
  1151→    );
  1152→
  1153→    final preview = <_ItAutoFillPreviewRow>[];
  1154→    for (var i = 0; i < solutionsToUse.length; i++) {
  1155→      final sourceTitle = solutionsToUse[i].title.trim();
  1156→      final title = _cleanSolutionTitle(sourceTitle).trim().isEmpty
  1157→          ? 'Solution ${i + 1}'
  1158→          : _cleanSolutionTitle(sourceTitle).trim();
  1159→      final suggestions = (generated[sourceTitle] ?? const <String>[])
  1160→          .map((entry) => entry.trim())
  1161→          .where((entry) => entry.isNotEmpty)
  1162→          .toList(growable: false);
  1163→      if (suggestions.isEmpty) continue;
  1164→      preview.add(_ItAutoFillPreviewRow(title: title, items: suggestions));
  1165→    }
  1166→    return preview;
  1167→  }

  1306→      );

  1867→  Future<void> _regenerateSingleTechField(
  1868→      TextEditingController controller, int index) async {
  1869→    if (index >= _solutions.length) return;
  1870→
  1871→    final provider = ProjectDataHelper.getProvider(context);
  1872→    final messenger = ScaffoldMessenger.of(context);
  1873→    try {
  1874→      final solution = _solutions[index];
  1875→      final solutionsToUse = [solution];
  1876→      final contextNotes = _notesController.text.trim();
  1877→
  1878→      final map = await _openAi.generateTechnologiesForSolutions(
  1879→        solutionsToUse,
  1880→        contextNotes: contextNotes,
  1881→      );
  1882→      if (!mounted) return;
  1883→
  1884→      final tech = map[solution.title] ?? const <String>[];
  1885→      controller.text = tech.isEmpty ? '' : tech.map((e) => '- $e').join('\n');
  1886→
  1887→      await provider.saveToFirebase(checkpoint: 'it_tech_field_regenerated');
  1888→
  1889→      if (mounted) {
  1890→        messenger.showSnackBar(
  1891→          const SnackBar(content: Text('IT tech field regenerated')),
  1892→        );
  1893→      }
  1894→    } catch (e) {
  1895→      if (mounted) {
  1896→        messenger.showSnackBar(
  1897→          SnackBar(content: Text('Failed to regenerate: $e')),
  1898→        );
  1899→      }
  1900→    }
  1901→  }
