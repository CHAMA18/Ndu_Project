    11→// Remove markdown bold markers commonly produced by the model (e.g. *text* or **text**)
    12→String _stripAsterisks(String s) => s.replaceAll('*', '');

    14→/// Strips markdown code fences (```json ... ```) and extracts raw JSON text.
    15→/// OpenAI may wrap JSON responses in markdown code blocks, so this ensures
    16→/// jsonDecode receives clean JSON.
    17→String _extractJson(String text) {
    18→  final codeBlockRegex = RegExp(r'```(?:json)?\s*\n?([\s\S]*?)\n?```');
    19→  final match = codeBlockRegex.firstMatch(text);
    20→  if (match != null) return match.group(1)?.trim() ?? text.trim();
    21→  final jsonStart = text.indexOf('{');
    22→  final jsonEnd = text.lastIndexOf('}');
    23→  if (jsonStart >= 0 && jsonEnd > jsonStart) return text.substring(jsonStart, jsonEnd + 1);
    24→  return text.trim();
    25→}

    69→/// Returns a human-readable USD-to-currency rate hint for AI prompts.
    70→String _usdRateHint(String currency) {
    71→  final rate = _usdToCurrencyRates[currency.toUpperCase()] ?? 1.0;
    72→  if (rate >= 100) return rate.toStringAsFixed(0);
    73→  if (rate >= 10) return rate.toStringAsFixed(1);
    74→  return rate.toStringAsFixed(2);
    75→}
    76→
    77→/// Returns a rough converted value hint for AI prompts.
    78→String _convertHint(double usdAmount, String currency) {
    79→  final rate = _usdToCurrencyRates[currency.toUpperCase()] ?? 1.0;
    80→  final converted = usdAmount * rate;
    81→  if (converted >= 1000000) return '${(converted / 1000000).toStringAsFixed(1)}M';
    82→  if (converted >= 1000) return converted.toStringAsFixed(0);
    83→  return converted.toStringAsFixed(converted % 1 == 0 ? 0 : 2);
    84→}

   171→class AiCostItem {
   172→  final String item;
   173→  final String description;
   174→  final double estimatedCost;
   175→  final double roiPercent; // percent value, e.g., 15.0 means 15%
   176→  final Map<int, double> npvByYear;
   177→  final double npv; // default to selected baseline (5-year when available)
   178→
   179→  AiCostItem({
   180→    required this.item,
   181→    required this.description,
   182→    required this.estimatedCost,
   183→    required this.roiPercent,
   184→    required Map<int, double> npvByYear,
   185→  })  : npvByYear = Map.unmodifiable({...npvByYear}),
   186→        npv =
   187→            npvByYear[5] ?? (npvByYear.isNotEmpty ? npvByYear.values.first : 0);
   188→
   189→  double npvForYear(int years) => npvByYear[years] ?? npv;
   190→
   191→  factory AiCostItem.fromMap(Map<String, dynamic> map) {
   192→    final Map<int, double> parsedNpvs = {};
   193→
   194→    double toD(v) {
   195→      if (v == null) return 0;
   196→      if (v is num) return v.toDouble();
   197→      final s = v.toString().replaceAll(',', '').replaceAll('%', '').trim();
   198→      return double.tryParse(s) ?? 0;
   199→    }
   200→
   201→    void addNpv(int year, dynamic value) {
   202→      final parsed = toD(value);
   203→      if (parsedNpvs.containsKey(year) || parsed == 0) return;
   204→      parsedNpvs[year] = parsed;
   205→    }
   206→
   207→    final npvField = map['npv'];
   208→    if (npvField is Map) {
   209→      for (final entry in npvField.entries) {
   210→        final key = entry.key.toString().replaceAll(RegExp(r'[^0-9]'), '');
   211→        final year = int.tryParse(key);
   212→        if (year != null) addNpv(year, entry.value);
   213→      }
   214→    } else {
   215→      addNpv(5, npvField);
   216→    }
   217→
   218→    final npvByYearsField = map['npv_by_years'];
   219→    if (npvByYearsField is Map) {
   220→      for (final entry in npvByYearsField.entries) {
   221→        final key = entry.key.toString().replaceAll(RegExp(r'[^0-9]'), '');
   222→        final year = int.tryParse(key);
   223→        if (year != null) addNpv(year, entry.value);
   224→      }
   225→    }
   226→
   227→    if (parsedNpvs.isEmpty) addNpv(5, 0);
   228→
   229→    return AiCostItem(
   230→      item: (map['item'] ?? map['project_item'] ?? '').toString().trim(),
   231→      description: (map['description'] ?? '').toString().trim(),
   232→      estimatedCost: toD(map['estimated_cost']),
   233→      roiPercent: toD(map['roi_percent']),
   234→      npvByYear: parsedNpvs,
   235→    );
   236→  }
   237→}

   467→class OpenAiServiceSecure {
   468→  final http.Client _client;
   469→  static const int maxRetries = 1;
   470→  static const Duration retryDelay = Duration(seconds: 1);
   471→  static const Duration _interRequestDelay = Duration(milliseconds: 120);
   472→  static Future<void> _serializedQueue = Future<void>.value();
   473→
   474→  OpenAiServiceSecure({http.Client? client})
   475→      : _client = client ?? http.Client();
   476→
   477→  Future<T> _runSerialized<T>(Future<T> Function() operation) {
   478→    final completer = Completer<T>();
   479→    _serializedQueue = _serializedQueue.catchError((_) {}).then((_) async {
   480→      try {
   481→        final result = await operation();
   482→        completer.complete(result);
   483→      } catch (error, stackTrace) {
   484→        completer.completeError(error, stackTrace);
   485→      } finally {
   486→        await Future<void>.delayed(_interRequestDelay);
   487→      }
   488→    });
   489→    return completer.future;
   490→  }

   492→  Future<String> generateCompletion(
   493→    String prompt, {
   494→    int maxTokens = 1200,
   495→    double temperature = 0.4,
   496→  }) async {
   497→    final trimmedPrompt = prompt.trim();
   498→    if (trimmedPrompt.isEmpty) return '';
   499→    if (!OpenAiConfig.isConfigured) {
   500→      throw const OpenAiNotConfiguredException();
   501→    }
   502→
   503→    return _runSerialized(() async {
   504→      final uri = OpenAiConfig.chatUri();
   505→      final headers = OpenAiConfig.headers();
   506→
   507→      final body = jsonEncode(OpenAiConfig.wrapBody({
   508→        'model': OpenAiConfig.model,
   509→        'temperature': temperature,
   510→        'max_completion_tokens': maxTokens,
