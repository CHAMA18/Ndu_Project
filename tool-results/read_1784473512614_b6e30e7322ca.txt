   107→  Future<void> _autoPopulateFromProjectData(BuildContext context) async {
   108→    final data = ProjectDataHelper.getData(context);
   109→    if (data.planningNotes['interface_auto_populated'] == 'true') return;
   110→    if (data.interfaceEntries.isNotEmpty) return;
   111→
   112→    final entries = <InterfaceEntry>[];
   113→    final now = DateTime.now().toIso8601String();
   114→    final logEntries = List<InterfaceChangeLogEntry>.from(data.interfaceChangeLog);
   115→
   116→    // From Vendors
   117→    for (final v in data.vendors) {
   118→      final name = v.name.trim();
   119→      if (name.isEmpty) continue;
   120→      entries.add(InterfaceEntry(
   121→        boundary: name,
   122→        interfaceType: 'Contractual',
   123→        partyA: 'Project Team',
   124→        partyB: name,
   125→        criticality: 'Major',
   126→        priority: 'Medium',
   127→        status: 'Pending',
   128→        notes: 'Vendor interface - ${v.equipmentOrService.trim()}',
   129→      ));
   130→    }
   131→
   132→    // From Contractors
   133→    for (final c in data.contractors) {
   134→      final name = c.name.trim();
   135→      if (name.isEmpty) continue;
   136→      entries.add(InterfaceEntry(
   137→        boundary: name,
   138→        interfaceType: 'Contractual',
   139→        partyA: 'Project Team',
   140→        partyB: name,
   141→        criticality: 'Major',
   142→        priority: 'Medium',
   143→        status: 'Pending',
   144→        notes: 'Contractor interface - ${c.service.trim()}',
   145→      ));
   146→    }
   147→
   148→    // From External Integrations (Technology Planning)
   149→    for (final ext in data.externalIntegrations) {
   150→      final name = ext['name']?.toString().trim() ?? '';
   151→      if (name.isEmpty) continue;
   152→      entries.add(InterfaceEntry(
   153→        boundary: name,
   154→        interfaceType: 'Technical',
   155→        partyA: 'Project Team',
   156→        partyB: name,
   157→        criticality: 'Major',
   158→        priority: 'Medium',
   159→        status: 'Pending',
   160→        owner: ext['description']?.toString() ?? '',
   161→        notes: 'Technical interface from Technology Planning',
   162→        id: '',
   163→      ));
   164→    }
   165→
   166→    if (entries.isEmpty) return;
   167→
   168→    for (final entry in entries) {
   169→      logEntries.add(InterfaceChangeLogEntry(
   170→        interfaceId: entry.id,
   171→        interfaceName: entry.boundary,
   172→        action: 'Auto-Populated',
   173→        newValue: entry.boundary,
   174→        changedAt: now,
   175→      ));
   176→    }
   177→
   178→    await ProjectDataHelper.updateAndSave(
   179→      context: context,
   180→      checkpoint: 'interface_management',
   181→      dataUpdater: (d) => d.copyWith(
   182→        planningNotes: {
   183→          ...d.planningNotes,
   184→          'interface_auto_populated': 'true',
   185→        },
   186→        interfaceEntries: [...d.interfaceEntries, ...entries],
   187→        interfaceChangeLog: [...d.interfaceChangeLog, ...logEntries],
   188→      ),
   189→      showSnackbar: true,
   190→    );
   191→
   192→    if (mounted) {
   193→      ScaffoldMessenger.of(context).showSnackBar(
   194→        SnackBar(
   195→          content: Text('Auto-populated ${entries.length} interface(s) from project data'),
   196→        ),
   197→      );
   198→    }
   199→    unawaited(ActivityLogService.instance.logCheckpointActivity(
   200→      projectId: data.projectId ?? '',
   201→      checkpoint: 'interface_management',
   202→      action: 'Interface Auto-Populated',
   203→      details: {'count': entries.length},
   204→    ));
   205→
   206→    // After basic auto-populate, kick off AI generation for additional
   207→    // cross-scope interfaces (vendors, contractors, other orgs, program/portfolio)
   208→    if (mounted) {
   209→      unawaited(_aiPopulateInterfaces(context, data));
   210→    }

   213→  /// AI-enhanced interface population. Calls generateInterfaceEntries to
   214→  /// identify cross-scope interfaces with vendors, contractors, other
   215→  /// companies/organizations, and program/portfolio projects. De-duplicates
   216→  /// against existing entries by boundary name.
   217→  /// 
   218→  /// Enhanced to include comprehensive project context:
   219→  /// - Stakeholder register data
   220→  /// - Vendor/contractor information from Procurement
   221→  /// - Existing interfaces already identified
   222→  /// - Company/organization info where available
   223→  Future<void> _aiPopulateInterfaces(
   224→      BuildContext context, ProjectDataModel data) async {
   225→    if (data.planningNotes['interface_ai_populated'] == 'true') return;
   226→
   227→    final additionalInterfaces =
   228→        data.planningNotes['additional_interfaces'] ?? 'Not Applicable';
   229→
   230→    // Build a comprehensive program/portfolio context including:
   231→    // - Stakeholder register data
   232→    // - Vendor/contractor info
   233→    // - Organization details
   234→    // - Existing interfaces
   235→    final programPortfolioContext = <String>[];
   236→    
   237→    try {
   238→      // 1. Stakeholder Management Plan summary from planning notes
   239→      final stakeholderPlan =
   240→          data.planningNotes['stakeholder_management_plan'] ??
   241→          data.planningNotes['stakeholder_engagement_plan'] ??
   242→          '';
   243→      if (stakeholderPlan.isNotEmpty) {
   244→        programPortfolioContext
   245→            .add('Stakeholder Management Plan: $stakeholderPlan');
   246→      }
   247→      
   248→      // 2. Stakeholder Register Entries - extract key stakeholders for interface analysis
   249→      if (data.stakeholderEntries.isNotEmpty) {
   250→        final stakeholderSummary = data.stakeholderEntries
   251→            .where((s) => (s.name ?? '').trim().isNotEmpty)
   252→            .take(15)
   253→            .map((s) => '- ${s.name} (${s.role ?? "Unknown Role"}): ${s.influence ?? ""} influence, ${s.interest ?? ""} interest')
   254→            .join('\n');
   255→        if (stakeholderSummary.isNotEmpty) {
   256→          programPortfolioContext.add('Key Stakeholders:\n$stakeholderSummary');
   257→        }
   258→      }
   259→      
   260→      // 3. Vendor Information from Procurement
   261→      if (data.vendors.isNotEmpty) {
   262→        final vendorSummary = data.vendors
   263→            .where((v) => v.name.trim().isNotEmpty)
   264→            .map((v) => '- ${v.name}: ${v.equipmentOrService} (${v.status})')
   265→            .join('\n');
   266→        if (vendorSummary.isNotEmpty) {
   267→          programPortfolioContext.add('Procurement Vendors:\n$vendorSummary');
   268→        }
   269→      }
   270→      
   271→      // 4. Contractor Information from Procurement
   272→      if (data.contractors.isNotEmpty) {
   273→        final contractorSummary = data.contractors
   274→            .where((c) => c.name.trim().isNotEmpty)
   275→            .map((c) => '- ${c.name}: ${c.service} (${c.status})')
   276→            .join('\n');
   277→        if (contractorSummary.isNotEmpty) {
   278→          programPortfolioContext.add('Contractors:\n$contractorSummary');
   279→        }
   280→      }
   281→      
   282→      // 5. Existing interfaces already identified (for de-duplication context)
   283→      if (data.interfaceEntries.isNotEmpty) {
   284→        final existingIfaces = data.interfaceEntries
   285→            .where((e) => e.boundary.trim().isNotEmpty)
   286→            .take(10)
   287→            .map((e) => '- ${e.boundary} [${e.interfaceType}] ${e.partyA} ↔ ${e.partyB}')
   288→            .join('\n');
   289→        if (existingIfaces.isNotEmpty) {
   290→          programPortfolioContext.add('Existing Interfaces Already Identified:\n$existingIfaces');
   291→        }
   292→      }
   293→      
   294→      // 6. External integrations from Technology Planning
   295→      if (data.externalIntegrations.isNotEmpty) {
   296→        final extIntegrations = data.externalIntegrations
   297→            .where((e) => (e['name']?.toString().trim() ?? '').isNotEmpty)
   298→            .map((e) => '- ${e["name"]}: ${e["description"] ?? ""} (${e["connectionType"] ?? "API"})')
   299→            .join('\n');
   300→        if (extIntegrations.isNotEmpty) {
   301→          programPortfolioContext.add('Technology External Integrations:\n$extIntegrations');
   302→        }
   303→      }
   304→      
   305→      // 7. Project site/location info if available
   306→      final siteInfo = data.planningNotes['project_site'] ?? 
   307→          data.planningNotes['site_location'] ?? 
   308→          data.planningNotes['location'] ?? '';
   309→      if (siteInfo.isNotEmpty) {
   310→        programPortfolioContext.add('Project Site/Location: $siteInfo');
   311→      }
   312→      
   313→    } catch (_) {
   314→      debugPrint('Error building program/portfolio context for AI interface generation');
   315→    }
   316→
   317→    final projectContext =
   318→        ProjectDataHelper.buildFepContext(data, sectionLabel: 'Interface Management');
   319→    if (projectContext.trim().isEmpty) return;
   320→
   321→    List<Map<String, dynamic>> aiEntries = [];
   322→    try {
   323→      aiEntries = await OpenAiServiceSecure().generateInterfaceEntries(
   324→        context: projectContext,
   325→        additionalInterfaces: additionalInterfaces,
   326→        programPortfolioContext: programPortfolioContext.join('\n\n'),
   327→        maxItems: 10,
   328→      );
   329→    } catch (e) {
   330→      debugPrint('AI interface generation failed: $e');
   331→      return;
   332→    }
   333→
   334→    if (!mounted || aiEntries.isEmpty) {
   335→      // Mark as attempted so we don't retry on every rebuild
   336→      await ProjectDataHelper.updateAndSave(
   337→        context: context,
   338→        checkpoint: 'interface_management',
   339→        showSnackbar: false,
   340→        dataUpdater: (d) => d.copyWith(
   341→          planningNotes: {
   342→            ...d.planningNotes,
   343→            'interface_ai_populated': 'true',
   344→          },
   345→        ),
   346→      );
   347→      return;
   348→    }
   349→
   350→    // De-duplicate against existing entries by boundary name (case-insensitive)
   351→    final existingBoundaries = data.interfaceEntries
   352→        .map((e) => e.boundary.trim().toLowerCase())
   353→        .where((s) => s.isNotEmpty)
   354→        .toSet();
   355→
   356→    final newEntries = <InterfaceEntry>[];
   357→    final logEntries = <InterfaceChangeLogEntry>[];
   358→    final now = DateTime.now().toIso8601String();
   359→
   360→    for (final m in aiEntries) {
   361→      final boundary = (m['boundary'] ?? '').toString().trim();
   362→      if (boundary.isEmpty) continue;
   363→      if (existingBoundaries.contains(boundary.toLowerCase())) continue;
   364→      newEntries.add(InterfaceEntry(
   365→        boundary: boundary,
   366→        interfaceType: (m['interfaceType'] ?? 'Organizational').toString(),
   367→        partyA: (m['partyA'] ?? 'Project Team').toString(),
   368→        partyB: (m['partyB'] ?? '').toString(),
   369→        criticality: (m['criticality'] ?? 'Major').toString(),
   370→        priority: (m['priority'] ?? 'Medium').toString(),
   371→        status: (m['status'] ?? 'Pending').toString(),
   372→        owner: (m['owner'] ?? '').toString(),
   373→        notes: (m['notes'] ?? '').toString(),
   374→      ));
   375→      existingBoundaries.add(boundary.toLowerCase());
   376→      logEntries.add(InterfaceChangeLogEntry(
   377→        interfaceId: '',
   378→        interfaceName: boundary,
   379→        action: 'AI-Generated',
   380→        newValue: boundary,
   381→        changedAt: now,
   382→      ));
   383→    }
   384→
   385→    if (newEntries.isEmpty) {
   386→      await ProjectDataHelper.updateAndSave(
   387→        context: context,
   388→        checkpoint: 'interface_management',
   389→        showSnackbar: false,
   390→        dataUpdater: (d) => d.copyWith(
   391→          planningNotes: {
   392→            ...d.planningNotes,
   393→            'interface_ai_populated': 'true',
   394→          },
   395→        ),
   396→      );
   397→      return;
   398→    }
   399→
   400→    await ProjectDataHelper.updateAndSave(
   401→      context: context,
   402→      checkpoint: 'interface_management',
   403→      showSnackbar: false,
   404→      dataUpdater: (d) => d.copyWith(
   405→        planningNotes: {
   406→          ...d.planningNotes,
   407→          'interface_ai_populated': 'true',
   408→        },
   409→        interfaceEntries: [...d.interfaceEntries, ...newEntries],
   410→        interfaceChangeLog: [...d.interfaceChangeLog, ...logEntries],
   411→      ),
   412→    );
   413→
   414→    if (!mounted) return;
   415→    ScaffoldMessenger.of(context).showSnackBar(
   416→      SnackBar(
   417→        content: Text(
   418→            'KAZ AI identified ${newEntries.length} additional cross-scope interface(s).'),
   419→        duration: const Duration(seconds: 4),
   420→      ),
   421→    );
   422→  }

   732→ Future<void> _generateAiPlan() async {
   733→   if (_isAiGenerating) return;
   734→   setState(() => _isAiGenerating = true);
   735→   try {
   736→     final data = ProjectDataHelper.getData(context);
   737→     final additionalInterfaces =
   738→         data.planningNotes['additional_interfaces'] ?? 'Not Applicable';
   739→     
   740→     // Build enhanced stakeholder plan context
   741→     final stakeholderPlanBuffer = StringBuffer();
   742→     final baseStakeholderPlan =
   743→         data.planningNotes['stakeholder_management_plan'] ??
   744→         data.planningNotes['stakeholder_engagement_plan'] ??
   745→         '';
   746→     if (baseStakeholderPlan.isNotEmpty) {
   747→       stakeholderPlanBuffer.writeln(baseStakeholderPlan);
   748→       stakeholderPlanBuffer.writeln();
   749→     }
   750→     
   751→     // Add vendor/contractor context for richer AI plan generation
   752→     if (data.vendors.isNotEmpty) {
   753→       stakeholderPlanBuffer.writeln('Procurement Vendors:');
   754→       for (final v in data.vendors.take(8)) {
   755→         if (v.name.trim().isNotEmpty) {
   756→           stakeholderPlanBuffer.writeln('- ${v.name}: ${v.equipmentOrService}');
   757→         }
   758→       }
   759→       stakeholderPlanBuffer.writeln();
   760→     }
   761→     
   762→     if (data.contractors.isNotEmpty) {
   763→       stakeholderPlanBuffer.writeln('Contractors:');
   764→       for (final c in data.contractors.take(6)) {
   765→         if (c.name.trim().isNotEmpty) {
   766→           stakeholderPlanBuffer.writeln('- ${c.name}: ${c.service}');
   767→         }
   768→       }
   769→       stakeholderPlanBuffer.writeln();
   770→     }
   771→     
   772→     // Add existing interface summary for context
   773→     if (data.interfaceEntries.isNotEmpty) {
   774→       stakeholderPlanBuffer.writeln('Current Interface Register Summary (${data.interfaceEntries.length} entries):');
   775→       final criticalIfaces = data.interfaceEntries.where((e) => e.criticality.toLowerCase() == 'critical').toList();
   776→       final majorIfaces = data.interfaceEntries.where((e) => e.criticality.toLowerCase() == 'major').toList();
   777→       if (criticalIfaces.isNotEmpty) {
   778→         stakeholderPlanBuffer.writeln('Critical Interfaces: ${criticalIfaces.map((e) => e.boundary).join(", ")}');
   779→       }
   780→       if (majorIfaces.isNotEmpty) {
   781→         stakeholderPlanBuffer.writeln('Major Interfaces: ${majorIfaces.map((e) => e.boundary).join(", ")}');
   782→       }
   783→       stakeholderPlanBuffer.writeln();
   784→     }
   785→     
   786→     final enhancedStakeholderPlan = stakeholderPlanBuffer.toString().trim();
   787→     
   788→     final projectContext = ProjectDataHelper.buildFepContext(
   789→         data, sectionLabel: 'Interface Management');

   791→       setState(() => _isAiGenerating = false);
   792→       return;
   793→     }

   795→       context: projectContext,
   796→       additionalInterfaces: additionalInterfaces,
   797→       stakeholderPlanSummary: enhancedStakeholderPlan,
   798→     );
   799→     if (!mounted) return;
   800→     if (plan.trim().isNotEmpty) {
   801→       setState(() => _initialText = plan.trim());
   802→       await ProjectDataHelper.updateAndSave(
   803→         context: context,
   804→         checkpoint: 'interface_management',
   805→         showSnackbar: false,
   806→         dataUpdater: (d) => d.copyWith(
   807→           planningNotes: {
   808→             ...d.planningNotes,
   809→             _noteKey: plan.trim(),
   810→             'interface_plan_ai_generated': 'true',
   811→           },
   812→         ),
   813→       );
   814→       if (mounted) {
   815→         setState(() => _lastSavedAt = DateTime.now());
   816→       }
   817→     }
   818→   } catch (e) {
   819→     debugPrint('AI Interface Management Plan generation failed: $e');
   820→   } finally {
   821→     if (mounted) setState(() => _isAiGenerating = false);
   822→   }
   823→ }
