   136→        interfaceType: 'Contractual',

   138→        partyB: name,

   211→  /// AI-enhanced interface population. Calls generateInterfaceEntries to
   212→  /// identify cross-scope interfaces with vendors, contractors, other
   213→  /// companies/organizations, and program/portfolio projects. De-duplicates
   214→  /// against existing entries by boundary name.
   215→  Future<void> _aiPopulateInterfaces(
   216→      BuildContext context, ProjectDataModel data) async {
   217→    if (data.planningNotes['interface_ai_populated'] == 'true') return;
   218→
   219→    final additionalInterfaces =
   220→        data.planningNotes['additional_interfaces'] ?? 'Not Applicable';
   221→
   222→    // Build a compact program/portfolio context from the user's other projects
   223→    final programPortfolioContext = <String>[];
   224→    try {
   225→      // The stakeholder plan summary is read from planning notes
   226→      final stakeholderPlan =
   227→          data.planningNotes['stakeholder_management_plan'] ??
   228→          data.planningNotes['stakeholder_engagement_plan'] ??
   229→          '';
   230→      if (stakeholderPlan.isNotEmpty) {
   231→        programPortfolioContext
   232→            .add('Stakeholder Management Plan: $stakeholderPlan');
   233→      }
   234→    } catch (_) {}
   235→
   236→    final projectContext =
   237→        ProjectDataHelper.buildFepContext(data, sectionLabel: 'Interface Management');
   238→    if (projectContext.trim().isEmpty) return;
   239→
   240→    List<Map<String, dynamic>> aiEntries = [];
   241→    try {
   242→      aiEntries = await OpenAiServiceSecure().generateInterfaceEntries(
   243→        context: projectContext,
   244→        additionalInterfaces: additionalInterfaces,
   245→        programPortfolioContext: programPortfolioContext.join('\n\n'),
   246→        maxItems: 10,
   247→      );
   248→    } catch (e) {
   249→      debugPrint('AI interface generation failed: $e');
   250→      return;
   251→    }
   252→
   253→    if (!mounted || aiEntries.isEmpty) {
   254→      // Mark as attempted so we don't retry on every rebuild
   255→      await ProjectDataHelper.updateAndSave(
   256→        context: context,
   257→        checkpoint: 'interface_management',
   258→        showSnackbar: false,
   259→        dataUpdater: (d) => d.copyWith(
   260→          planningNotes: {
   261→            ...d.planningNotes,
   262→            'interface_ai_populated': 'true',
   263→          },
   264→        ),
   265→      );
   266→      return;
   267→    }
   268→
   269→    // De-duplicate against existing entries by boundary name (case-insensitive)
   270→    final existingBoundaries = data.interfaceEntries
   271→        .map((e) => e.boundary.trim().toLowerCase())
   272→        .where((s) => s.isNotEmpty)
   273→        .toSet();
   274→
   275→    final newEntries = <InterfaceEntry>[];
   276→    final logEntries = <InterfaceChangeLogEntry>[];
   277→    final now = DateTime.now().toIso8601String();
   278→
   279→    for (final m in aiEntries) {
   280→      final boundary = (m['boundary'] ?? '').toString().trim();
   281→      if (boundary.isEmpty) continue;
   282→      if (existingBoundaries.contains(boundary.toLowerCase())) continue;
   283→      newEntries.add(InterfaceEntry(
   284→        boundary: boundary,
   285→        interfaceType: (m['interfaceType'] ?? 'Organizational').toString(),
   286→        partyA: (m['partyA'] ?? 'Project Team').toString(),
   287→        partyB: (m['partyB'] ?? '').toString(),
   288→        criticality: (m['criticality'] ?? 'Major').toString(),
   289→        priority: (m['priority'] ?? 'Medium').toString(),
   290→        status: (m['status'] ?? 'Pending').toString(),
   291→        owner: (m['owner'] ?? '').toString(),
   292→        notes: (m['notes'] ?? '').toString(),
   293→      ));
   294→      existingBoundaries.add(boundary.toLowerCase());
   295→      logEntries.add(InterfaceChangeLogEntry(
   296→        interfaceId: '',
   297→        interfaceName: boundary,
   298→        action: 'AI-Generated',
   299→        newValue: boundary,
   300→        changedAt: now,
   301→      ));
   302→    }
   303→
   304→    if (newEntries.isEmpty) {
   305→      await ProjectDataHelper.updateAndSave(
   306→        context: context,
   307→        checkpoint: 'interface_management',
   308→        showSnackbar: false,
   309→        dataUpdater: (d) => d.copyWith(
   310→          planningNotes: {
   311→            ...d.planningNotes,
   312→            'interface_ai_populated': 'true',
   313→          },
   314→        ),
   315→      );
   316→      return;
   317→    }
   318→
   319→    await ProjectDataHelper.updateAndSave(
   320→      context: context,
   321→      checkpoint: 'interface_management',
   322→      showSnackbar: false,
   323→      dataUpdater: (d) => d.copyWith(
   324→        planningNotes: {
   325→          ...d.planningNotes,
   326→          'interface_ai_populated': 'true',
   327→        },
   328→        interfaceEntries: [...d.interfaceEntries, ...newEntries],
   329→        interfaceChangeLog: [...d.interfaceChangeLog, ...logEntries],
   330→      ),
   331→    );
   332→
   333→    if (!mounted) return;
   334→    ScaffoldMessenger.of(context).showSnackBar(
   335→      SnackBar(
   336→        content: Text(
   337→            'KAZ AI identified ${newEntries.length} additional cross-scope interface(s).'),
   338→        duration: const Duration(seconds: 4),
   339→      ),
   340→    );
   341→  }

   629→ // Auto-generate the AI plan if empty and the popup has been completed

  1376→void _syncInterfaceRiskToGlobal({
  1377→  required BuildContext context,
  1378→  required InterfaceEntry entry,
  1379→  InterfaceEntry? oldEntry,
  1380→}) {
  1381→  final data = ProjectDataHelper.getData(context);
  1382→  var riskItems = List<ExecutionRiskItem>.from(data.executionRiskItems);
  1383→  final existingIdx = riskItems.indexWhere((r) => r.id == 'iface_${entry.id}');
  1384→  final hasNewRisk = entry.risk.trim().isNotEmpty;
  1385→
  1386→  if (!hasNewRisk) {
  1387→    if (existingIdx != -1) {
  1388→      riskItems.removeAt(existingIdx);
  1389→    }
  1390→  } else {
  1391→    final newItem = _interfaceEntryToRiskItem(entry);
  1392→    if (existingIdx != -1) {
  1393→      riskItems[existingIdx] = newItem;
  1394→    } else {
  1395→      riskItems.add(newItem);
  1396→    }
  1397→  }
  1398→
  1399→  if (existingIdx != -1 || hasNewRisk) {
  1400→    ProjectDataHelper.updateAndSave(
  1401→      context: context,
  1402→      checkpoint: 'interface_management',
  1403→      dataUpdater: (d) => d.copyWith(executionRiskItems: riskItems),
  1404→      showSnackbar: false,
  1405→    );
  1406→  }
  1407→
  1408→  // Also sync to the central FEP Risk Register (bidirectional)
  1409→  if (hasNewRisk) {
  1410→    unawaited(ProjectDataHelper.upsertRiskToRegister(
  1411→      context: context,
  1412→      sourceSection: 'Interface Management',
  1413→      riskName: 'Interface: ${entry.boundary}',
  1414→      description: entry.notes.isNotEmpty
  1415→          ? entry.notes
  1416→          : 'Risk associated with interface ${entry.boundary} (${entry.partyA} ↔ ${entry.partyB})',
  1417→      category: 'Interface Management',
  1418→      impactLevel: entry.criticality.toLowerCase() == 'critical'
  1419→          ? 'High'
  1420→          : entry.criticality.toLowerCase() == 'major'
  1421→              ? 'Medium'
  1422→              : 'Low',
  1423→      likelihood: 'Medium',
  1424→      mitigationStrategy: entry.risk,
  1425→      discipline: entry.interfaceType,
  1426→      owner: entry.owner,
  1427→      status: entry.status,
  1428→    ));
  1429→  } else {
  1430→    unawaited(ProjectDataHelper.removeRiskFromRegister(
  1431→      context: context,
  1432→      sourceSection: 'Interface Management',
  1433→      riskName: 'Interface: ${entry.boundary}',
  1434→    ));
  1435→  }
  1436→
  1437→  // Log to central activities log
  1438→  unawaited(ProjectDataHelper.logActivityToCentral(
  1439→    context: context,
  1440→    sourceSection: 'Interface Management',
  1441→    title: 'Interface ${entry.boundary} — ${hasNewRisk ? "Risk identified" : "Risk cleared"}',
  1442→    description: 'Interface between ${entry.partyA} and ${entry.partyB}. Risk: ${entry.risk}',
  1443→    phase: 'Planning',
  1444→    discipline: entry.interfaceType,
  1445→    role: entry.owner,
  1446→  ));
  1447→}

  1449→void _removeInterfaceRiskFromGlobal({
  1450→  required BuildContext context,
  1451→  required String interfaceId,
  1452→}) {
  1453→  final data = ProjectDataHelper.getData(context);
  1454→  final riskItems = List<ExecutionRiskItem>.from(data.executionRiskItems);
  1455→  // Find the interface entry to get its boundary name for central register cleanup
  1456→  final entry = data.interfaceEntries.firstWhere(
  1457→    (e) => e.id == interfaceId,
  1458→    orElse: () => InterfaceEntry(boundary: ''),
  1459→  );
  1460→  riskItems.removeWhere((r) => r.id == 'iface_$interfaceId');
  1461→  ProjectDataHelper.updateAndSave(
  1462→    context: context,
  1463→    checkpoint: 'interface_management',
  1464→    dataUpdater: (d) => d.copyWith(executionRiskItems: riskItems),
  1465→    showSnackbar: false,
  1466→  );
  1467→  // Also remove from central FEP Risk Register
  1468→  if (entry.boundary.isNotEmpty) {
  1469→    unawaited(ProjectDataHelper.removeRiskFromRegister(
  1470→      context: context,
  1471→      sourceSection: 'Interface Management',
  1472→      riskName: 'Interface: ${entry.boundary}',
  1473→    ));
  1474→  }
