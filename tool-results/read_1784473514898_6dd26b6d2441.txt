   195→  // ── Load saved state (visited tabs, stakeholder confirmation, plans, SDGs) ──
   196→  void _loadSavedState() {
   197→    final ssherData = ProjectDataHelper.getData(context).ssherData;
   198→    setState(() {
   199→      _stakeholderConfirmed = ssherData.stakeholderConfirmed;
   200→      _autoSyncToCostEstimate = ssherData.autoSyncToCostEstimate;
   201→      // Restore visited tabs from persistent storage
   202→      for (final cat in _SsherCategory.values) {
   203→        if (cat == _SsherCategory.cost) continue;
   204→        if (ssherData.categoryApplicability[cat.name] == false) {
   205→          _visitedTabs.add(cat);
   206→        }
   207→        if (ssherData.tabsVisited[cat.name] == true) {
   208→          _visitedTabs.add(cat);
   209→        }
   210→      }
   211→      // Restore per-category plans
   212→      for (final entry in ssherData.categoryPlans.entries) {
   213→        if (entry.value.trim().isNotEmpty) {
   214→          _categoryPlans[entry.key] = entry.value;
   215→          _categoryPlanLoaded[entry.key] = true;
   216→        }
   217→      }
   218→    });
   219→    // If auto-sync is enabled, mark all existing entries as already-synced
   220→    // so we don't retroactively push them on the first save.
   221→    if (_autoSyncToCostEstimate) {
   222→      for (final e in ssherData.entries) {
   223→        _autoSyncedEntryIds.add(e.id);
   224→      }
   225→    }
   226→  }

   228→  Future<void> _saveTabsVisited() async {
   229→    final visitedMap = <String, bool>{};
   230→    for (final cat in _SsherCategory.values) {
   231→      if (cat == _SsherCategory.cost) continue;
   232→      visitedMap[cat.name] = _visitedTabs.contains(cat);
   233→    }
   234→    await ProjectDataHelper.updateAndSave(
   235→      context: context,
   236→      checkpoint: 'ssher',
   237→      showSnackbar: false,
   238→      dataUpdater: (data) => data.copyWith(
   239→        ssherData: data.ssherData.copyWith(tabsVisited: visitedMap),
   240→      ),
   241→    );
   242→  }

   340→  /// Pull the Initiation phase (Front End Planning) security section data
   341→  /// into the Security tab so the user has continuity. Only runs once.
   342→  void _pullInitiationSecurityData() {
   343→    if (_initiationSecurityPulled) return;
   344→    final data = ProjectDataHelper.getData(context);
   345→    final fep = data.frontEndPlanning;
   346→    final securityText = fep.security.trim();
   347→
   348→    // Pull the security notes as a single entry on the Security tab if we don't already have one
   349→    final hasSecurityEntry = _securityEntries.any((e) =>
   350→        e.concern.toLowerCase().contains('initiation') ||
   351→        e.notes.toLowerCase().contains('initiation security') ||
   352→        e.concern == 'Initiation Phase Security Plan');
   353→
   354→    if (securityText.isNotEmpty && !hasSecurityEntry) {
   355→      // Summarize the initiation security text into a 200-char entry
   356→      final summary = securityText.length > 280
   357→          ? '${securityText.substring(0, 280)}...'
   358→          : securityText;
   359→      final entry = SsherEntry(
   360→        category: 'security',
   361→        department: 'IT Security',
   362→        teamMember: 'Security Analyst (carried from Initiation)',
   363→        concern: 'Initiation Phase Security Plan (carried forward)',
   364→        riskLevel: 'High',
   365→        mitigation: summary,
   366→        notes: 'Auto-imported from Front End Planning – Security section.',
   367→      );
   368→      setState(() {
   369→        _securityEntries.insert(0, entry);
   370→        _initiationSecurityPulled = true;
   371→      });
   372→      _saveEntries();
   373→    } else if (hasSecurityEntry) {
   374→      _initiationSecurityPulled = true;
   375→    }
   376→  }

   378→  Future<void> _populateSsherEntriesFromAi() async {
   379→    if (_entriesGenerated || _isGeneratingEntries) return;
   380→    if (_allEntries().isNotEmpty) {
   381→      _entriesGenerated = true;
   382→      return;
   383→    }
   384→
   385→    final projectData = ProjectDataHelper.getData(context);
   386→    final contextText =
   387→        ProjectDataHelper.buildFepContext(projectData, sectionLabel: 'SSHER');
   388→    if (contextText.trim().isEmpty) {
   389→      _entriesGenerated = true;
   390→      return;
   391→    }
   392→
   393→    setState(() => _isGeneratingEntries = true);
   394→
   395→    List<SsherEntry> generatedEntries = [];
   396→    try {
   397→      generatedEntries = await OpenAiServiceSecure()
   398→          .generateSsherEntries(context: contextText, itemsPerCategory: 2);
   399→    } catch (error) {
   400→      debugPrint('SSHER entries AI call failed: $error');
   401→    }
   402→
   403→    if (!mounted) return;
   404→
   405→    if (_allEntries().isNotEmpty) {
   406→      setState(() => _isGeneratingEntries = false);
   407→      _entriesGenerated = true;
   408→      return;
   409→    }
   410→
   411→    final safety = <SsherEntry>[];
   412→    final security = <SsherEntry>[];
   413→    final health = <SsherEntry>[];
   414→    final environment = <SsherEntry>[];
   415→    final regulatory = <SsherEntry>[];
   416→
   417→    for (final entry in generatedEntries) {
   418→      switch (entry.category) {
   419→        case 'safety':
   420→          safety.add(entry);
   421→          break;
   422→        case 'security':
   423→          security.add(entry);
   424→          break;
   425→        case 'health':
   426→          health.add(entry);
   427→          break;
   428→        case 'environment':
   429→          environment.add(entry);
   430→          break;
   431→        case 'regulatory':
   432→          regulatory.add(entry);
   433→          break;
   434→      }
   435→    }
   436→
   437→    setState(() {
   438→      _safetyEntries = safety;
   439→      _securityEntries = security;
   440→      _healthEntries = health;
   441→      _environmentEntries = environment;
   442→      _regulatoryEntries = regulatory;
   443→      _isGeneratingEntries = false;
   444→    });
   445→    _entriesGenerated = true;
   446→    await _saveEntries();
   447→    // After AI generation, also pull in Initiation security data
   448→    _pullInitiationSecurityData();
   449→  }

   451→  /// Ensures the per-category AI plan is loaded (lazy-loaded on first tab visit)
   452→  Future<void> _ensureCategoryPlanLoaded(_SsherCategory cat) async {
   453→    if (cat == _SsherCategory.cost) return;
   454→    final key = cat.name;
   455→    if (_categoryPlanLoaded[key] == true) return;
   456→    if (_categoryPlans[key]?.isNotEmpty == true) {
   457→      _categoryPlanLoaded[key] = true;
   458→      return;
   459→    }
   460→    if (_categoryPlanLoading[key] == true) return;
   461→
   462→    setState(() => _categoryPlanLoading[key] = true);
   463→
   464→    final projectData = ProjectDataHelper.getData(context);
   465→    final contextText =
   466→        ProjectDataHelper.buildFepContext(projectData, sectionLabel: 'SSHER');
   467→    if (contextText.trim().isEmpty) {
   468→      setState(() {
   469→        _categoryPlanLoaded[key] = true;
   470→        _categoryPlanLoading[key] = false;
   471→      });
   472→      return;
   473→    }
   474→
   475→    String plan = '';
   476→    try {
   477→      plan = await OpenAiServiceSecure().generateSsherCategoryPlan(
   478→        context: contextText,
   479→        category: key,
   480→      );
   481→    } catch (e) {
   482→      debugPrint('generateSsherCategoryPlan failed for $key: $e');
   483→    }
   484→
   485→    if (!mounted) return;
   486→
   487→    final trimmed = plan.trim();
   488→    setState(() {
   489→      if (trimmed.isNotEmpty) {
   490→        _categoryPlans[key] = trimmed;
   491→      }
   492→      _categoryPlanLoaded[key] = true;
   493→      _categoryPlanLoading[key] = false;
   494→    });
   495→
   496→    if (trimmed.isNotEmpty) {
   497→      await ProjectDataHelper.updateAndSave(
   498→        context: context,
   499→        checkpoint: 'ssher',
   500→        showSnackbar: false,
   501→        dataUpdater: (data) => data.copyWith(
   502→          ssherData: data.ssherData.copyWith(
   503→            categoryPlans: Map<String, String>.from(_categoryPlans),
   504→          ),
   505→        ),
   506→      );
   507→    }
   508→  }

   519→  /// Generates SDG recommendations (for the Environment tab)
   520→  Future<void> _loadSdgRecommendations() async {
   521→    if (_sdgsLoaded || _isGeneratingSdgs) return;
   522→    setState(() => _isGeneratingSdgs = true);
   523→
   524→    final projectData = ProjectDataHelper.getData(context);
   525→    final contextText =
   526→        ProjectDataHelper.buildFepContext(projectData, sectionLabel: 'SSHER');
   527→    if (contextText.trim().isEmpty) {
   528→      setState(() {
   529→        _isGeneratingSdgs = false;
   530→        _sdgsLoaded = true;
   531→      });
   532→      return;
   533→    }
   534→
   535→    List<Map<String, String>> recs = [];
   536→    try {
   537→      recs = await OpenAiServiceSecure()
   538→          .generateSdgRecommendations(context: contextText);
   539→    } catch (e) {
   540→      debugPrint('generateSdgRecommendations failed: $e');
   541→    }
   542→
   543→    if (!mounted) return;
   544→    setState(() {
   545→      _sdgRecommendations = recs;
   546→      _isGeneratingSdgs = false;
   547→      _sdgsLoaded = true;
   548→    });
   549→
   550→    if (recs.isNotEmpty) {
   551→      final sdgList = recs.map((r) => r['goal'] ?? '').where((g) => g.isNotEmpty).toList();
   552→      await ProjectDataHelper.updateAndSave(
   553→        context: context,
   554→        checkpoint: 'ssher',
   555→        showSnackbar: false,
   556→        dataUpdater: (data) => data.copyWith(
   557→          ssherData: data.ssherData.copyWith(applicableSdgs: sdgList),
   558→        ),
   559→      );
   560→    }
   561→  }

   563→  List<SsherEntry> _entriesForCategory(_SsherCategory category) {

   605→  /// Sync high/critical SSHER entries to the global execution risk log.
   606→  /// This ensures that significant SSHER concerns are visible in the central risk register.
   607→  Future<void> _syncSsherRisksToGlobalLog() async {
   608→    try {
   609→      final allEntries = _allEntries();
   610→      
   611→      // Only sync entries with elevated risk levels (not 'Low' or empty)
   612→      final riskyEntries = allEntries.where((e) {
   613→        final level = e.riskLevel.toLowerCase();
   614→        return level.isNotEmpty && 
   615→               level != 'low' && 
   616→               level != 'very low' &&
   617→               e.concern.trim().isNotEmpty;
   618→      }).toList();
   619→      
   620→      for (final entry in riskyEntries) {
   621→        final riskItem = RiskSyncService.ssherEntryToRiskItem(entry);
   622→        await RiskSyncService.syncRiskToGlobalLog(
   623→          context: context,
   624→          sectionId: 'ssher',
   625→          sectionLabel: 'SSHER (${entry.category.toUpperCase()})',
   626→          riskItem: riskItem,
   627→          idPrefix: 'ssher_',
   628→        );
   629→      }
   630→      
   631→      debugPrint('SSHER: Synced ${riskyEntries.length} risky entries to global risk log');
   632→    } catch (e) {
   633→      debugPrint('SSHER: Error syncing risks to global log: $e');
   634→    }
   635→  }

   644→      );
   645→    } catch (e) {
   646→      debugPrint('SSHER: Error removing risk from global log: $e');
   647→    }
   648→  }
